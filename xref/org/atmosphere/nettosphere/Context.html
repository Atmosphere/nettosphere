117     private Set _getResourcePaths(String path, boolean resolve) {
118         File a;
119         if (resolve) {
120             a = new File(b.basePath + path);
121         } else {
122             a = new File(path);
123         }
124 
125         File[] files = a.listFiles();
126         Set<String> s = new HashSet<String>();
127         try {
128             if (files != null) {
129                 for (File f : files) {
130                     if (f.isDirectory()) {
131                         Set inner = _getResourcePaths(f.getAbsolutePath(), false);
132                         s.addAll(inner);
133                     } else {
134                         s.add(f.getCanonicalPath().substring(b.basePath.length()));
135                     }
136                 }
137             }
138         } catch (IOException e) {
139             logger.trace("", e);
140         }
141         return s;
142     }
