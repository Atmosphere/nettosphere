16  package org.atmosphere.nettosphere;
17  
18  import org.jboss.netty.channel.ChannelPipeline;
19  import org.jboss.netty.channel.ChannelPipelineFactory;
20  import org.jboss.netty.handler.codec.http.HttpChunkAggregator;
21  import org.jboss.netty.handler.codec.http.HttpRequestDecoder;
22  import org.jboss.netty.handler.codec.http.HttpResponseEncoder;
23  
24  import static org.jboss.netty.channel.Channels.pipeline;
25  
26  class AtmosphereChannelPipelineFactory implements
27          ChannelPipelineFactory {
28  
29      private final transient NettyAtmosphereHandler nettyAtmosphereHandler;
30  
31      public AtmosphereChannelPipelineFactory(final NettyAtmosphereHandler nettyAtmosphereHandler) {
32          this.nettyAtmosphereHandler = nettyAtmosphereHandler;
33      }
34  
35      /**
36       * Retrieve the channel pipeline factory.
37       *
38       * @see org.jboss.netty.channel.ChannelPipelineFactory#getPipeline()
39       */
40      public ChannelPipeline getPipeline() {
41          final ChannelPipeline pipeline = pipeline();
42          pipeline.addLast("decoder", new HttpRequestDecoder());
43          pipeline.addLast("aggregator", new HttpChunkAggregator(65536));
44          pipeline.addLast("encoder", new HttpResponseEncoder());
45          pipeline.addLast("nettyAtmosphereHandler", nettyAtmosphereHandler);
46          return pipeline;
47      }
48  
49  }
