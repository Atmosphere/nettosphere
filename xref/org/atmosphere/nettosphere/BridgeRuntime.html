293     @Override
294     public void channelRead(final ChannelHandlerContext ctx, final Object messageEvent) throws URISyntaxException, IOException {
295         try {
296             handleMessageEvent(ctx, messageEvent);
297         } finally {
298              if (messageEvent instanceof ReferenceCounted) {
299                 ReferenceCounted refMsg = (ReferenceCounted) messageEvent;
300                 if(refMsg.refCnt() > 0)
301                     refMsg.release();
302             }
303         }
304     }

306     private void handleMessageEvent(final ChannelHandlerContext ctx, final Object messageEvent) throws URISyntaxException, IOException {
307         Object msg = messageEvent;
308 
309         if (isShutdown.get()) {
310             ctx.channel().close().addListener(ChannelFutureListener.CLOSE);
311             return;
312         }
313 
314         if (msg instanceof HttpRequest) {
315             HttpRequest r = HttpRequest.class.cast(msg);
316             // Netty fail to decode headers separated by a ','
317             List<String> c = r.headers().getAll("Connection");
318             String u = r.headers().get("Upgrade");
319             boolean webSocket = false;
320             if (u != null && u.equalsIgnoreCase("websocket")) {
321                 webSocket = true;
322             }
323 
324             for (String connection : c) {
325                 if (connection != null && connection.toLowerCase().equalsIgnoreCase("upgrade")) {
326                     webSocket = true;
327                 }
328             }
329 
330             logger.trace("Handling request {}", r);
331             if (webSocket) {
332                 handleWebSocketHandshake(ctx, messageEvent);
333             } else {
334                 handleHttp(ctx, messageEvent);
335             }
336         } else if (msg instanceof WebSocketFrame) {
337             handleWebSocketFrame(ctx, messageEvent);
338         } else if (msg instanceof HttpChunkedInput) {
339             handleHttp(ctx, messageEvent);
340         }
341     }

343     private void handleWebSocketHandshake(final ChannelHandlerContext ctx, Object messageEvent) throws IOException, URISyntaxException {
344         final HttpRequest request = (HttpRequest) messageEvent;
345 
346         // Allow only GET methods.
347         if (request.getMethod() != GET) {
348             sendHttpResponse(ctx, request, new DefaultHttpResponse(HTTP_1_1, FORBIDDEN));
349             return;
350         }
351 
352         ctx.pipeline().addBefore(BridgeRuntime.class.getName(), "encoder", new HttpResponseEncoder());
353         WebSocketServerHandshakerFactory wsFactory = new WebSocketServerHandshakerFactory(getWebSocketLocation(request),
354                 config.subProtocols(),
355                 false,
356                 maxWebSocketFrameSize);
357 
358         WebSocketServerHandshaker handshaker = wsFactory.newHandshaker(request);
359 
360         if (handshaker == null) {
361             wsFactory.sendUnsupportedVersionResponse(ctx.channel());
362         } else {
363             final NettyWebSocket webSocket = new NettyWebSocket(ctx.channel(), framework.getAtmosphereConfig(), config.noInternalAlloc(), config.binaryWrite());
364             final AtmosphereRequest atmosphereRequest = createAtmosphereRequest(ctx, request, EMPTY);
365 
366             if (!webSocketProcessor.handshake(atmosphereRequest)) {
367                 sendError(ctx, HttpResponseStatus.BAD_REQUEST, null);
368                 return;
369             }
370 
371             webSocketProcessor.notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent("", HANDSHAKE, webSocket));
372 
373             handshaker.handshake(ctx.channel(), (FullHttpRequest) request).addListener(new ChannelFutureListener() {
374                 @Override
375                 public void operationComplete(ChannelFuture future) throws Exception {
376                     if (!future.isSuccess()) {
377                         future.channel().close();
378                     } else {
379                         websocketChannels.add(ctx.channel());
380 
381                         ctx.channel().attr(ATTACHMENT).set(webSocket);
382                         if (config.noInternalAlloc()) {
383                             webSocket.resource(proxiedResource);
384                         }
385 
386                         AtmosphereResponse response = config.noInternalAlloc() ? proxiedResponse :
387                                 AtmosphereResponseImpl.newInstance(framework.getAtmosphereConfig(), atmosphereRequest, webSocket);
388                         webSocketProcessor.open(webSocket, atmosphereRequest, response);
389 
390                         if (webSocketTimeout > 0) {
391                             webSocket.closeFuture(suspendTimer.scheduleAtFixedRate(new Runnable() {
392                                 @Override
393                                 public void run() {
394                                     if (webSocket.lastWriteTimeStampInMilliseconds() != 0 && (System.currentTimeMillis() - webSocket.lastWriteTimeStampInMilliseconds() > webSocketTimeout)) {
395                                         logger.debug("Timing out {}", webSocket);
396                                         webSocket.close();
397                                     }
398                                 }
399                             }, webSocketTimeout, webSocketTimeout, TimeUnit.MILLISECONDS));
400                         }
401                     }
402                 }
403             });
404         }
405     }

407     private void handleWebSocketFrame(final ChannelHandlerContext ctx, final Object messageEvent) throws URISyntaxException, IOException {
408         WebSocketFrame frame = (WebSocketFrame) messageEvent;
409 
410         logger.trace("Received frame {}", frame.getClass().getName());
411 
412         // Check for closing frame
413         WebSocket attachment = WebSocket.class.cast(ctx.channel().attr(ATTACHMENT).get());
414         ByteBuf binaryData = frame.content();
415         byte[] body = null;
416         if (binaryData.isReadable()) {
417             body = new byte[binaryData.readableBytes()];
418             binaryData.readBytes(body);
419         }
420         if (frame instanceof CloseWebSocketFrame) {
421             ctx.channel().write(frame).addListener(ChannelFutureListener.CLOSE);
422         } else if (frame instanceof PingWebSocketFrame) {
423             if (WebSocketPingPongListener.class.isAssignableFrom(webSocketProcessor.getClass())) {
424                 WebSocketPingPongListener.class.cast(webSocketProcessor).onPing(attachment, body, 0, body.length);
425             } else {
426                 ctx.channel().writeAndFlush(new PongWebSocketFrame(binaryData));
427             }
428         } else if (frame instanceof BinaryWebSocketFrame || (frame instanceof TextWebSocketFrame && config.textFrameAsBinary())) {
429             webSocketProcessor.invokeWebSocketProtocol(attachment, body, 0, body.length);
430         } else if (frame instanceof TextWebSocketFrame) {
431             webSocketProcessor.invokeWebSocketProtocol(attachment, ((TextWebSocketFrame) frame).text());
432         } else if (frame instanceof PongWebSocketFrame) {
433 
434             if (WebSocketPingPongListener.class.isAssignableFrom(webSocketProcessor.getClass())) {
435                 WebSocketPingPongListener.class.cast(webSocketProcessor).onPong(attachment, body, 0, body.length);
436             }
437 
438             if (config.enablePong()) {
439                 ctx.channel().writeAndFlush(new PingWebSocketFrame(frame.content()));
440             } else {
441                 logger.trace("Received Pong Frame on Channel {}", ctx.channel());
442             }
443         } else {
444             logger.warn("{} frame types not supported", frame.getClass());
445             ctx.channel().close();
446         }
447     }

449     private AtmosphereRequest createAtmosphereRequest(final ChannelHandlerContext ctx, final HttpRequest request, byte[] body) throws URISyntaxException, UnsupportedEncodingException, MalformedURLException {
450         final String base = getBaseUri(request);
451         final URI requestUri = new URI(base.substring(0, base.length() - 1) + request.getUri());
452         final String ct = HttpHeaders.getHeader(request, "Content-Type", "text/plain");
453         final long cl = HttpHeaders.getContentLength(request, 0);
454         String method = request.getMethod().name();
455 
456         String queryString = requestUri.getQuery();
457         Map<String, String[]> qs = new HashMap<String, String[]>();
458         if (queryString != null) {
459             parseQueryString(qs, queryString);
460         }
461 
462         if (ct.equalsIgnoreCase("application/x-www-form-urlencoded")) {
463             if (FullHttpRequest.class.isAssignableFrom(request.getClass())) {
464                 parseQueryString(qs, new String(body));
465             }
466         }
467 
468         String u = requestUri.toURL().toString();
469         int last = u.indexOf("?") == -1 ? u.length() : u.indexOf("?");
470         String url = u.substring(0, last);
471         int l;
472 
473         if (url.contains(config.mappingPath())) {
474             l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3 + config.mappingPath().length();
475         } else {
476             l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3;
477         }
478 
479         HttpSession session = null;
480         if (framework.getAtmosphereConfig().isSupportSession()) {
481             String[] transport = qs.get(HeaderConfig.X_ATMOSPHERE_TRANSPORT);
482             if (transport != null && transport.length > 0) {
483                 String[] uuid = qs.get(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
484                 if (uuid != null && uuid.length > 0) {
485                     // TODO: Session is only supported until an unsubscribe is received.
486                     if (transport[0].equalsIgnoreCase(HeaderConfig.DISCONNECT_TRANSPORT_MESSAGE)) {
487                         sessions.remove(uuid[0]);
488                     } else {
489                         session = sessions.get(uuid[0]);
490 
491                         if (session == null) {
492                             session = new FakeHttpSession("-1", null, System.currentTimeMillis(), -1);
493                         }
494                     }
495                 }
496             }
497         }
498 
499         final Map<String, Object> attributes = new HashMap<String, Object>();
500         AtmosphereRequestImpl.Builder requestBuilder = new AtmosphereRequestImpl.Builder();
501         requestBuilder.requestURI(url.substring(l))
502                 .requestURL(url)
503                 .pathInfo(url.substring(l))
504                 .headers(getHeaders(request))
505                 .method(method)
506                 .contentType(ct)
507                 .contentLength(cl)
508                 // We need to read attribute after doComet
509                 .destroyable(false)
510                 .attributes(attributes)
511                 .servletPath(config.mappingPath())
512                 .session(session)
513                 .cookies(getCookies(request))
514                 .queryStrings(qs)
515                 .remoteInetSocketAddress(new Callable<InetSocketAddress>() {
516                     @Override
517                     public InetSocketAddress call() throws Exception {
518                         return (InetSocketAddress) ctx.channel().remoteAddress();
519                     }
520                 })
521                 .localInetSocketAddress(new Callable<InetSocketAddress>() {
522 
523                     @Override
524                     public InetSocketAddress call() throws Exception {
525                         return (InetSocketAddress) ctx.channel().localAddress();
526                     }
527                 });
528 
529         if (body.length > 0) {
530             requestBuilder.body(body);
531         }
532 
533         return requestBuilder.build();
534 
535     }

537     private void handleHttp(final ChannelHandlerContext ctx, final Object messageEvent) throws URISyntaxException, IOException {
538 
539         boolean skipClose = false;
540         AtmosphereResponse response = null;
541         AtmosphereRequest request = null;
542         Action a = null;
543         boolean resumeOnBroadcast = false;
544         boolean keptOpen = false;
545         ChannelWriter asyncWriter = null;
546         String method = "GET";
547         boolean writeHeader = false;
548         boolean forceSuspend = false;
549         boolean aggregateBodyInMemory = config.aggregateRequestBodyInMemory();
550 
551         try {
552             if (messageEvent instanceof HttpRequest) {
553                 final HttpRequest hrequest = (HttpRequest) messageEvent;
554 
555                 byte[] body = EMPTY;
556                 if (FullHttpRequest.class.isAssignableFrom(messageEvent.getClass())) {
557                     ByteBuf b = FullHttpRequest.class.cast(messageEvent).content();
558                     if (b.isReadable()) {
559                         body = new byte[b.readableBytes()];
560                         b.readBytes(body);
561                     }
562                 }
563 
564                 // First let's try to see if it's a static resources
565                 if (!hrequest.getUri().contains(HeaderConfig.X_ATMOSPHERE)) {
566                     try {
567                         hrequest.headers().add(STATIC_MAPPING, "true");
568                         super.channelRead(ctx, messageEvent);
569 
570                         if (HttpHeaders.getHeader(hrequest, SERVICED) != null) {
571                             return;
572                         }
573                     } catch (Exception e) {
574                         logger.debug("Unexpected State", e);
575                     } finally {
576                         hrequest.headers().set(STATIC_MAPPING, "false");
577                     }
578                 }
579 
580                 boolean ka = HttpHeaders.isKeepAlive(hrequest);
581                 asyncWriter = config.supportChunking() ?
582                         new ChunkedWriter(ctx.channel(), true, ka) :
583                         new StreamWriter(ctx.channel(), true, ka);
584 
585                 method = hrequest.getMethod().name();
586 
587                 request = createAtmosphereRequest(ctx, hrequest, body);
588                 request.setAttribute(KEEP_ALIVE, new Boolean(ka));
589 
590                 // Hacky. Is the POST doesn't contains a body, we must not close the connection yet.
591                 AtmosphereRequestImpl.Body b = request.body();
592                 if (!aggregateBodyInMemory
593                         && !hrequest.getMethod().equals(GET)
594                         && !b.isEmpty()
595                         && (b.hasString() && b.asString().isEmpty())
596                         || (b.hasBytes() && b.asBytes().length == 0)) {
597                     forceSuspend = true;
598                 }
599             } else {
600                 request = State.class.cast(ctx.attr(ATTACHMENT).get()).request;
601                 boolean isLast = HttpChunkedInput.class.cast(messageEvent).isEndOfInput();
602                 Boolean ka = (Boolean) request.getAttribute(KEEP_ALIVE);
603 
604                 asyncWriter = config.supportChunking() ?
605                         new ChunkedWriter(ctx.channel(), isLast, ka) :
606                         new StreamWriter(ctx.channel(), isLast, ka);
607                 method = request.getMethod();
608                 ByteBuf internalBuffer = HttpChunkedInput.class.cast(messageEvent).readChunk(ctx).content();
609 
610                 if (!aggregateBodyInMemory && internalBuffer.hasArray()) {
611                     request.body(internalBuffer.array());
612                 } else {
613                     logger.trace("Unable to read in memory the request's bytes. Using stream");
614                     request.body(new ByteBufInputStream(internalBuffer));
615                 }
616 
617                 if (!isLast) {
618                     forceSuspend = true;
619                 }
620             }
621 
622             response = new AtmosphereResponseImpl.Builder()
623                     .asyncIOWriter(asyncWriter)
624                     .writeHeader(writeHeader)
625                     .destroyable(false)
626                     .header("Connection", "Keep-Alive")
627                     .header("Server", "Nettosphere/3.0")
628                     .request(request).build();
629 
630             if (config.supportChunking()) {
631                 response.setHeader("Transfer-Encoding", "chunked");
632             }
633 
634             a = framework.doCometSupport(request, response);
635             if (forceSuspend) {
636                 a.type(Action.TYPE.SUSPEND);
637                 // leave the stream open
638                 keptOpen = true;
639             }
640 
641             String transport = (String) request.getAttribute(FrameworkConfig.TRANSPORT_IN_USE);
642             if (transport == null) {
643                 transport = request.getHeader(X_ATMOSPHERE_TRANSPORT);
644             }
645 
646             if (a.type() == Action.TYPE.SUSPEND) {
647                 if (transport != null && (transport.equalsIgnoreCase(HeaderConfig.STREAMING_TRANSPORT)
648                         || transport.equalsIgnoreCase(SSE_TRANSPORT))) {
649                     keptOpen = true;
650                 } else if (transport != null && (
651                         transport.equalsIgnoreCase(HeaderConfig.LONG_POLLING_TRANSPORT) ||
652                                 transport.equalsIgnoreCase(HeaderConfig.JSONP_TRANSPORT))) {
653                     resumeOnBroadcast = true;
654                 }
655             }
656 
657             final Action action = (Action) request.getAttribute(NettyCometSupport.SUSPEND);
658             final State state = new State(request, action == null ? Action.CONTINUE : action);
659 
660             ctx.attr(ATTACHMENT).set(state);
661 
662             if (action != null && action.type() == Action.TYPE.SUSPEND) {
663                 if (action.timeout() != -1) {
664                     final AtomicReference<ChannelWriter> w = new AtomicReference<ChannelWriter>(asyncWriter);
665                     final AtomicReference<Future<?>> f = new AtomicReference<Future<?>>();
666                     f.set(suspendTimer.scheduleAtFixedRate(new Runnable() {
667                         @Override
668                         public void run() {
669                             if (!w.get().isClosed() && (System.currentTimeMillis() - w.get().lastTick()) > action.timeout()) {
670                                 AtmosphereResourceImpl impl = state.resource();
671                                 if (impl != null) {
672                                     asynchronousProcessor.endRequest(impl, false);
673                                     f.get().cancel(true);
674                                 }
675                             }
676                         }
677                     }, action.timeout(), action.timeout(), TimeUnit.MILLISECONDS));
678                 }
679             } else if (action != null && action.type() == Action.TYPE.RESUME) {
680                 resumeOnBroadcast = false;
681             }
682         } catch (AtmosphereMappingException ex) {
683             if (method.equalsIgnoreCase("GET")) {
684                 logger.trace("Unable to map the request {}, trying static file", messageEvent);
685             }
686         } catch (Throwable e) {
687             logger.error("Unable to process request", e);
688             throw new IOException(e);
689         } finally {
690             try {
691                 if (asyncWriter != null && !resumeOnBroadcast && !keptOpen) {
692                     if (!skipClose && response != null) {
693                         asyncWriter.close(response);
694                     } else {
695                         httpChannels.add(ctx.channel());
696                     }
697                 }
698             } finally {
699                 if (request != null && a != null && a.type() != Action.TYPE.SUSPEND) {
700                     request.destroy();
701                     response.destroy();
702                     framework.notify(Action.TYPE.DESTROYED, request, response);
703                 }
704             }
705         }
706     }

708     @Override
709     public void sendError(ChannelHandlerContext ctx, HttpResponseStatus status, FullHttpRequest e) {
710         // For websocket, we can't send an error
711         logger.trace("Error {} for {}", status, e);
712         if (websocketChannels.contains(ctx.channel())) {
713             logger.debug("Error {} for {}", status, e);
714             ctx.channel().close().addListener(ChannelFutureListener.CLOSE);
715         } else if (e != null) {
716             if (HttpHeaders.getHeader(e, STATIC_MAPPING, "false").equalsIgnoreCase("false")) {
717                 super.sendError(ctx, status, e);
718             }
719         } else {
720             super.sendError(ctx, status, e);
721         }
722     }

725         isShutdown.set(true);
726         if (framework != null) framework.destroy();
727 
728         httpChannels.close();
729         websocketChannels.write(new CloseWebSocketFrame());
730         suspendTimer.shutdown();

733     @Override
734     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
735         super.channelInactive(ctx);
736         Object o = ctx.attr(ATTACHMENT).get();
737 
738         if (o == null) return;
739 
740         if (WebSocket.class.isAssignableFrom(o.getClass())) {
741             NettyWebSocket webSocket = NettyWebSocket.class.cast(o);
742             logger.trace("Closing {}", webSocket.uuid());
743 
744             try {
745                 if (webSocket.closeFuture() != null) {
746                     webSocket.closeFuture().cancel(true);
747                 }
748 
749                 webSocketProcessor.close(webSocket, 1005);
750             } catch (Exception ex) {
751                 logger.error("{}", webSocket, ex);
752             }
753         } else if (State.class.isAssignableFrom(o.getClass())) {
754             logger.trace("State {}", o);
755             State s = State.class.cast(o);
756             if (s.action.type() == Action.TYPE.SUSPEND) {
757                 asynchronousProcessor.endRequest(s.resource(), true);
758             }
759         } else {
760             logger.error("Invalid state {} and Channel {}", o, ctx.channel());
761         }
762     }

764     @Override
765     public void exceptionCaught(ChannelHandlerContext ctx, Throwable e) {
766         // Ignore Disconnect exception.
767         if (e.getCause() != null
768                 && (e.getCause().getClass().equals(ClosedChannelException.class)
769                 || e.getCause().getClass().equals(IOException.class))) {
770             logger.trace("Exception", e.getCause());
771         } else if (e.getCause() != null && e.getCause().getClass().equals(TooLongFrameException.class)) {
772             logger.error("TooLongFrameException. The request will be closed, make sure you increase the Config.maxChunkContentLength() to a higher value.", e.getCause());
773             super.exceptionCaught(ctx, e);
774         } else {
775             logger.debug("Exception", e.getCause());
776             super.exceptionCaught(ctx, e);
777         }
778     }

796     private void parseQueryString(Map<String, String[]> qs, String queryString) {
797         if (queryString != null) {
798             String[] s = queryString.split("&");
799             for (String a : s) {
800                 String[] q = a.split("=");
801                 String[] z = new String[]{q.length > 1 ? q[1] : ""};
802                 qs.put(q[0], z);
803             }
804         }

807     private Set<javax.servlet.http.Cookie> getCookies(final HttpRequest request) {
808         Set<javax.servlet.http.Cookie> result = new HashSet<javax.servlet.http.Cookie>();
809         String cookieHeader = request.headers().get("Cookie");
810         if (cookieHeader != null) {
811             Set<io.netty.handler.codec.http.cookie.Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieHeader);
812             for (io.netty.handler.codec.http.cookie.Cookie cookie : cookies) {
813                 javax.servlet.http.Cookie c = new javax.servlet.http.Cookie(cookie.name(), cookie.value());
814 
815                 if (cookie.domain() != null) {
816                     c.setDomain(cookie.domain());
817                 }
818 
819                 c.setHttpOnly(cookie.isHttpOnly());
820                 c.setMaxAge((int) cookie.maxAge());
821                 if (cookie.path() != null) {
822                     c.setPath(cookie.path());
823                 }
824 
825                 c.setSecure(cookie.isSecure());
826                 result.add(c);
827 
828             }
829         }
830         return result;
831     }

837     private void sendHttpResponse(ChannelHandlerContext ctx, HttpRequest req, HttpResponse res) {
838         // Generate an error page if response status code is not OK (200).
839         if (res.getStatus().code() != 200) {
840             FullHttpResponse response = (FullHttpResponse) res;
841             response.content().writeBytes(Unpooled.copiedBuffer(res.getStatus().toString(), CharsetUtil.UTF_8));
842             setContentLength(res, response.content().readableBytes());
843         }
844 
845         // Send the response and close the connection if necessary.
846         ChannelFuture f = ctx.channel().write(res);
847         if (!isKeepAlive(req) || res.getStatus().code() != 200) {
848             f.addListener(ChannelFutureListener.CLOSE);
849         }
850     }
