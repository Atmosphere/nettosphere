141     public BridgeRuntime(final Config config) {
142         super(config.path());
143         this.config = config;
144         framework = new AtmosphereFramework();
145 
146         if (config.broadcaster() != null) {
147             framework.setDefaultBroadcasterClassName(config.broadcaster().getName());
148         }
149 
150         framework.setAtmosphereDotXmlPath(config.configFile());
151 
152         try {
153             if (config.broadcasterFactory() != null) {
154                 framework.setBroadcasterFactory(config.broadcasterFactory());
155             }
156         } catch (Throwable t) {
157             logger.trace("", t);
158         }
159 
160         if (config.broadcasterCache() != null) {
161             try {
162                 framework.setBroadcasterCacheClassName(config.broadcasterCache().getName());
163             } catch (Throwable t) {
164                 logger.trace("", t);
165             }
166         }
167 
168         Map<String, AtmosphereHandler> handlersMap = config.handlersMap();
169         for (Map.Entry<String, AtmosphereHandler> e : handlersMap.entrySet()) {
170             framework.addAtmosphereHandler(e.getKey(), e.getValue());
171         }
172 
173         final Map<String, WebSocketHandler> webSocketHandlerMap = config.webSocketHandlersMap();
174 
175         if (handlersMap.size() == 0 && !webSocketHandlerMap.isEmpty()) {
176             framework.addAtmosphereHandler(Broadcaster.ROOT_MASTER, REFLECTOR_ATMOSPHEREHANDLER);
177         }
178 
179         framework.getAtmosphereConfig().startupHook(new AtmosphereConfig.StartupHook() {
180             @Override
181             public void started(AtmosphereFramework framework) {
182                 for (Map.Entry<String, WebSocketHandler> e : webSocketHandlerMap.entrySet()) {
183                     framework.addWebSocketHandler(e.getKey(), e.getValue());
184                 }
185             }
186         });
187 
188         if (config.webSocketProtocol() != null) {
189             framework.setWebSocketProtocolClassName(config.webSocketProtocol().getName());
190         }
191 
192         for (AtmosphereInterceptor i : config.interceptors()) {
193             framework.interceptor(i);
194         }
195 
196         if (!config.scanPackages().isEmpty()) {
197             for (Class<?> s : config.scanPackages()) {
198                 framework.addAnnotationPackage(s);
199             }
200         }
201 
202         final Context context = new Context.Builder().attributes(config.servletContextAttributes()).contextPath(config.mappingPath()).basePath(config.path()).build();
203         ServletContext ctx = (ServletContext) Proxy.newProxyInstance(BridgeRuntime.class.getClassLoader(), new Class[]{ServletContext.class},
204                 new InvocationHandler() {
205                     @Override
206                     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
207                         try {
208                             Method stub = Context.class.getMethod(method.getName(), method.getParameterTypes());
209                             if (stub != null) {
210                                 return stub.invoke(context, args);
211                             } else {
212                                 logger.trace("Method {} not supported", method.getName());
213                                 return null;
214                             }
215                         } catch (NoSuchMethodException ex) {
216                             logger.trace("Method {} not supported", method.getName());
217                             return null;
218                         }
219                     }
220                 });
221 
222         try {
223             framework.externalizeDestroy(true).init(new NettyServletConfig(config.initParams(), ctx));
224         } catch (ServletException e) {
225             throw new RuntimeException(e);
226         }
227 
228         int max = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME, -1);
229         if (max != -1) {
230             webSocketTimeout = max;
231         }
232 
233         framework.setAsyncSupport(new NettyCometSupport(framework.getAtmosphereConfig()) {
234             @Override
235             public Action suspended(AtmosphereRequest request, AtmosphereResponse response) throws IOException, ServletException {
236                 Action a = super.suspended(request, response);
237                 if (framework.getAtmosphereConfig().isSupportSession()) {
238                     AtmosphereResource r = request.resource();
239                     HttpSession s = request.getSession(true);
240                     if (s != null) {
241                         sessions.put(r.uuid(), request.getSession(true));
242                     }
243                 }
244                 return a;
245             }
246 
247             @Override
248             public String toString() {
249                 return "NettoSphereAsyncSupport";
250             }
251         });
252 
253         suspendTimer = new ScheduledThreadPoolExecutor(Runtime.getRuntime().availableProcessors());
254         webSocketProcessor = WebSocketProcessorFactory.getDefault().getWebSocketProcessor(framework);
255 
256         if (config.supportChunking()) {
257             channelBufferPool = new ChannelBufferPool(0, config.writeBufferPoolSize(),
258                     config.writeBufferPoolCleanupFrequency(), framework.getAtmosphereConfig());
259         } else {
260             // Dangerous
261             channelBufferPool = null;
262         }
263 
264         for (String s : config.excludedInterceptors()) {
265             framework.excludeInterceptor(s);
266         }
267         asynchronousProcessor = AsynchronousProcessor.class.cast(framework.getAsyncSupport());
268         maxWebSocketFrameSize = config.maxWebSocketFrameSize();
269 
270         if (config.noInternalAlloc()) {
271             proxiedRequest = new AtmosphereRequest.Builder().build();
272             proxiedResponse = new AtmosphereResponse.Builder().build();
273             proxiedResource = new AtmosphereResourceImpl();
274         } else {
275             proxiedRequest = null;
276             proxiedResponse = null;
277             proxiedResource = null;
278         }

285     @Override
286     public void messageReceived(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
287         Object msg = messageEvent.getMessage();
288 
289         if (isShutdown.get()) {
290             ctx.getChannel().close().addListener(ChannelFutureListener.CLOSE);
291             return;
292         }
293 
294         if (msg instanceof HttpRequest) {
295             HttpRequest r = HttpRequest.class.cast(msg);
296             // Netty fail to decode headers separated by a ','
297             List<String> c = r.headers().getAll("Connection");
298             String u = r.headers().get("Upgrade");
299             boolean webSocket = false;
300             if (u != null && u.equalsIgnoreCase("websocket")) {
301                 webSocket = true;
302             }
303 
304             for (String connection : c) {
305                 if (connection != null && connection.toLowerCase().equalsIgnoreCase("upgrade")) {
306                     webSocket = true;
307                 }
308             }
309 
310             logger.trace("Handling request {}", r);
311             if (webSocket) {
312                 handleWebSocketHandshake(ctx, messageEvent);
313             } else {
314                 handleHttp(ctx, messageEvent);
315             }
316         } else if (msg instanceof WebSocketFrame) {
317             handleWebSocketFrame(ctx, messageEvent);
318         } else if (msg instanceof HttpChunk) {
319             handleHttp(ctx, messageEvent);
320         }
321     }

323     private void handleWebSocketHandshake(final ChannelHandlerContext ctx, MessageEvent messageEvent) throws IOException, URISyntaxException {
324         final HttpRequest request = (HttpRequest) messageEvent.getMessage();
325 
326         // Allow only GET methods.
327         if (request.getMethod() != GET) {
328             sendHttpResponse(ctx, request, new DefaultHttpResponse(HTTP_1_1, FORBIDDEN));
329             return;
330         }
331 
332         WebSocketServerHandshakerFactory wsFactory = new WebSocketServerHandshakerFactory(getWebSocketLocation(request),
333                 config.subProtocols(),
334                 false,
335                 maxWebSocketFrameSize);
336 
337         WebSocketServerHandshaker handshaker = wsFactory.newHandshaker(request);
338 
339         if (handshaker == null) {
340             wsFactory.sendUnsupportedWebSocketVersionResponse(ctx.getChannel());
341         } else {
342 
343             final NettyWebSocket webSocket = new NettyWebSocket(ctx.getChannel(), framework.getAtmosphereConfig(), config.noInternalAlloc(), config.binaryWrite());
344             final AtmosphereRequest atmosphereRequest
345                     = createAtmosphereRequest(ctx, request);
346 
347             if (!webSocketProcessor.handshake(atmosphereRequest)) {
348                 sendError(ctx, BAD_REQUEST, null);
349                 return;
350             }
351 
352             webSocketProcessor.notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent("", HANDSHAKE, webSocket));
353 
354             handshaker.handshake(ctx.getChannel(), request).addListener(new ChannelFutureListener() {
355                 @Override
356                 public void operationComplete(ChannelFuture future) throws Exception {
357                     if (!future.isSuccess()) {
358                         future.getChannel().close();
359                     } else {
360                         websocketChannels.add(ctx.getChannel());
361 
362                         ctx.getChannel().setAttachment(webSocket);
363                         if (request.headers().contains("X-wakeUpNIO")) {
364                             /**
365                              * https://github.com/AsyncHttpClient/async-http-client/issues/471
366                              *
367                              * Netty 3.9.x has an issue and is unable to detect the websocket frame that will be produced if an AtmosphereInterceptor
368                              * like the JavaScriptProtocol write bytes just after the handshake's header. The effect is the message is lost when Netty decode the Handshake
369                              * request. This can be easily reproduced when wAsync is used with NettoSphere as server. For example, the following message
370                              *
371                              T 127.0.0.1:8080 -> 127.0.0.1:51295 [AP]
372                              HTTP/1.1 101 Switching Protocols.
373                              Upgrade: websocket.
374                              Connection: Upgrade.
375                              Sec-WebSocket-Accept: mFFTAW8KVZebToQFZZcFVWmJh8Y=.
376                              .
377 
378 
379                              T 127.0.0.1:8080 -> 127.0.0.1:51295 [AP]
380                              .21808c569-099b-4d8f-b657-c5965df40449|1391270901601
381 
382                              can be lost because the Netty's Decoder fail to realize the handshake contained a response's body. The error doesn't occurs all the time but under
383                              load happens more easily.
384                              */
385                             // Wake up the remote NIO Selector so Netty don't read the hanshake and the first message in a single read.
386                             for (int i = 0; i < 512; i++) {
387                                 webSocket.write(" ");
388                             }
389                         }
390 
391                         if (config.noInternalAlloc()) {
392                             webSocket.resource(proxiedResource);
393                         }
394 
395                         AtmosphereResponse response = config.noInternalAlloc() ? proxiedResponse :
396                                 AtmosphereResponse.newInstance(framework.getAtmosphereConfig(), atmosphereRequest, webSocket);
397                         webSocketProcessor.open(webSocket, atmosphereRequest, response);
398 
399                         if (webSocketTimeout > 0) {
400                             webSocket.closeFuture(suspendTimer.scheduleAtFixedRate(new Runnable() {
401                                 @Override
402                                 public void run() {
403                                     if (webSocket.lastWriteTimeStampInMilliseconds() != 0 && (System.currentTimeMillis() - webSocket.lastWriteTimeStampInMilliseconds() > webSocketTimeout)) {
404                                         logger.debug("Timing out {}", webSocket);
405                                         webSocket.close();
406                                     }
407                                 }
408                             }, webSocketTimeout, webSocketTimeout, TimeUnit.MILLISECONDS));
409                         }
410                     }
411                 }
412             });
413         }
414     }

416     private void handleWebSocketFrame(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
417         WebSocketFrame frame = (WebSocketFrame) messageEvent.getMessage();
418 
419         logger.trace("Received frame {}", frame.getClass().getName());
420 
421         // Check for closing frame
422         Object attachment = ctx.getChannel().getAttachment();
423         if (frame instanceof CloseWebSocketFrame) {
424             ctx.getChannel().write(frame).addListener(ChannelFutureListener.CLOSE);
425         } else if (frame instanceof PingWebSocketFrame) {
426             ChannelBuffer binaryData = frame.getBinaryData();
427             if (WebSocketPingPongListener.class.isAssignableFrom(webSocketProcessor.getClass())) {
428                 WebSocketPingPongListener.class.cast(webSocketProcessor).onPing((WebSocket) attachment, binaryData.array(), binaryData.arrayOffset(), binaryData.readableBytes());
429             } else {
430                 ctx.getChannel().write(new PongWebSocketFrame(frame.getBinaryData()));
431             }
432         } else if (frame instanceof BinaryWebSocketFrame || (frame instanceof TextWebSocketFrame && config.textFrameAsBinary())) {
433             ChannelBuffer binaryData = frame.getBinaryData();
434             webSocketProcessor.invokeWebSocketProtocol((WebSocket) attachment, binaryData.array(), binaryData.arrayOffset(), binaryData.readableBytes());
435         } else if (frame instanceof TextWebSocketFrame) {
436             webSocketProcessor.invokeWebSocketProtocol((WebSocket) attachment, ((TextWebSocketFrame) frame).getText());
437         } else if (frame instanceof PongWebSocketFrame) {
438             ChannelBuffer binaryData = frame.getBinaryData();
439 
440             if (WebSocketPingPongListener.class.isAssignableFrom(webSocketProcessor.getClass())) {
441                 WebSocketPingPongListener.class.cast(webSocketProcessor).onPong((WebSocket) attachment, binaryData.array(), binaryData.arrayOffset(), binaryData.readableBytes());
442             }
443 
444             if (config.enablePong()) {
445                 ctx.getChannel().write(new PingWebSocketFrame(frame.getBinaryData()));
446             } else {
447                 logger.trace("Received Pong Frame on Channel {}", ctx.getChannel());
448             }
449         } else {
450             logger.warn("{} frame types not supported", frame.getClass());
451             ctx.getChannel().close();
452         }
453     }

455     private AtmosphereRequest createAtmosphereRequest(final ChannelHandlerContext ctx, final HttpRequest request) throws URISyntaxException, UnsupportedEncodingException, MalformedURLException {
456         if (config.noInternalAlloc()) {
457             return proxiedRequest;
458         }
459 
460         final String base = getBaseUri(request);
461         final URI requestUri = new URI(base.substring(0, base.length() - 1) + request.getUri());
462         final String ct = HttpHeaders.getHeader(request, "Content-Type", "text/plain");
463         final long cl = HttpHeaders.getContentLength(request, 0);
464         String method = request.getMethod().getName();
465 
466         String queryString = requestUri.getQuery();
467         Map<String, String[]> qs = new HashMap<String, String[]>();
468         if (queryString != null) {
469             parseQueryString(qs, queryString);
470         }
471 
472         if (ct.equalsIgnoreCase("application/x-www-form-urlencoded")) {
473             parseQueryString(qs, new String(request.getContent().array(), "UTF-8"));
474         }
475 
476         String u = requestUri.toURL().toString();
477         int last = u.indexOf("?") == -1 ? u.length() : u.indexOf("?");
478         String url = u.substring(0, last);
479         int l;
480 
481         if (url.contains(config.mappingPath())) {
482             l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3 + config.mappingPath().length();
483         } else {
484             l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3;
485         }
486 
487         HttpSession session = null;
488         if (framework.getAtmosphereConfig().isSupportSession()) {
489             String[] transport = qs.get(HeaderConfig.X_ATMOSPHERE_TRANSPORT);
490             if (transport != null && transport.length > 0) {
491                 String[] uuid = qs.get(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
492                 if (uuid != null && uuid.length > 0) {
493                     // TODO: Session is only supported until an unsubscribe is received.
494                     if (transport[0].equalsIgnoreCase(HeaderConfig.DISCONNECT_TRANSPORT_MESSAGE)) {
495                         sessions.remove(uuid[0]);
496                     } else {
497                         session = sessions.get(uuid[0]);
498 
499                         if (session == null) {
500                             session = new FakeHttpSession("-1", null, System.currentTimeMillis(), -1);
501                         }
502                     }
503                 }
504             }
505         }
506 
507         final Map<String, Object> attributes = new HashMap<String, Object>();
508         AtmosphereRequest.Builder requestBuilder = new AtmosphereRequest.Builder();
509         requestBuilder.requestURI(url.substring(l))
510                 .requestURL(url)
511                 .pathInfo(url.substring(l))
512                 .headers(getHeaders(request))
513                 .method(method)
514                 .contentType(ct)
515                 .contentLength(cl)
516                         // We need to read attribute after doComet
517                 .destroyable(false)
518                 .attributes(attributes)
519                 .servletPath(config.mappingPath())
520                 .session(session)
521                 .cookies(getCookies(request))
522                 .queryStrings(qs)
523                 .remoteInetSocketAddress(new Callable<InetSocketAddress>() {
524                     @Override
525                     public InetSocketAddress call() throws Exception {
526                         return (InetSocketAddress) ctx.getChannel().getRemoteAddress();
527                     }
528                 })
529                 .localInetSocketAddress(new Callable<InetSocketAddress>() {
530 
531                     @Override
532                     public InetSocketAddress call() throws Exception {
533                         return (InetSocketAddress) ctx.getChannel().getLocalAddress();
534                     }
535                 });
536 
537         ChannelBuffer internalBuffer = request.getContent();
538         if (!config.aggregateRequestBodyInMemory() && !method.equalsIgnoreCase("GET")) {
539             return requestBuilder.body(internalBuffer.array()).build();
540         } else {
541             logger.trace("Unable to read in memory the request's bytes. Using stream");
542             return requestBuilder.inputStream(new ChannelBufferInputStream(internalBuffer)).build();
543         }
544     }

546     private void handleHttp(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
547 
548         boolean skipClose = false;
549         AtmosphereResponse response = null;
550         AtmosphereRequest request = null;
551         Action a = null;
552         boolean resumeOnBroadcast = false;
553         boolean keptOpen = false;
554         ChannelWriter asyncWriter = null;
555         String method = "GET";
556         boolean writeHeader = false;
557         boolean forceSuspend = false;
558         boolean aggregateBodyInMemory = config.aggregateRequestBodyInMemory();
559 
560         try {
561             if (messageEvent.getMessage() instanceof HttpRequest) {
562                 final HttpRequest hrequest = (HttpRequest) messageEvent.getMessage();
563                 boolean ka = HttpHeaders.isKeepAlive(hrequest);
564                 asyncWriter = config.supportChunking() ?
565                         new ChunkedWriter(ctx.getChannel(), true, ka, channelBufferPool) :
566                         new StreamWriter(ctx.getChannel(), true, ka);
567 
568                 method = hrequest.getMethod().getName();
569 
570                 // First let's try to see if it's a static resources
571                 if (!hrequest.getUri().contains(HeaderConfig.X_ATMOSPHERE)) {
572                     try {
573                         hrequest.headers().add(STATIC_MAPPING, "true");
574                         super.messageReceived(ctx, messageEvent);
575 
576                         if (HttpHeaders.getHeader(hrequest, SERVICED) != null) {
577                             return;
578                         }
579                     } catch (Exception e) {
580                         logger.debug("Unexpected State", e);
581                     } finally {
582                         hrequest.headers().set(STATIC_MAPPING, "false");
583                     }
584                 }
585 
586                 request = createAtmosphereRequest(ctx, hrequest);
587                 request.setAttribute(KEEP_ALIVE, new Boolean(ka));
588 
589                 // Hacky. Is the POST doesn't contains a body, we must not close the connection yet.
590                 AtmosphereRequest.Body b = request.body();
591                 if (!aggregateBodyInMemory
592                         && !hrequest.getMethod().equals(GET)
593                         && !b.isEmpty()
594                         && (b.hasString() && b.asString().isEmpty())
595                         || (b.hasBytes() && b.asBytes().length == 0)) {
596                     forceSuspend = true;
597                 }
598             } else {
599                 request = State.class.cast(ctx.getChannel().getAttachment()).request;
600                 boolean isLast = HttpChunk.class.cast(messageEvent.getMessage()).isLast();
601                 Boolean ka = (Boolean) request.getAttribute(KEEP_ALIVE);
602 
603                 asyncWriter = config.supportChunking() ?
604                         new ChunkedWriter(ctx.getChannel(), isLast, ka, channelBufferPool) :
605                         new StreamWriter(ctx.getChannel(), isLast, ka);
606                 method = request.getMethod();
607                 ChannelBuffer internalBuffer = HttpChunk.class.cast(messageEvent.getMessage()).getContent();
608 
609                 if (!aggregateBodyInMemory && internalBuffer.hasArray()) {
610                     request.body(internalBuffer.array());
611                 } else {
612                     logger.trace("Unable to read in memory the request's bytes. Using stream");
613                     request.body(new ChannelBufferInputStream(internalBuffer));
614                 }
615 
616                 if (!isLast) {
617                     forceSuspend = true;
618                 }
619             }
620 
621             response = new AtmosphereResponse.Builder()
622                     .asyncIOWriter(asyncWriter)
623                     .writeHeader(writeHeader)
624                     .destroyable(false)
625                     .header("Connection", "Keep-Alive")
626                     .header("Server", "Nettosphere/2.0")
627                     .request(request).build();
628 
629             if (config.supportChunking()) {
630                 response.setHeader("Transfer-Encoding", "chunked");
631             }
632 
633             a = framework.doCometSupport(request, response);
634             if (forceSuspend) {
635                 a.type(Action.TYPE.SUSPEND);
636                 // leave the stream open
637                 keptOpen = true;
638             }
639 
640             String transport = (String) request.getAttribute(FrameworkConfig.TRANSPORT_IN_USE);
641             if (transport == null) {
642                 transport = request.getHeader(X_ATMOSPHERE_TRANSPORT);
643             }
644 
645             if (a.type() == Action.TYPE.SUSPEND) {
646                 if (transport != null && (transport.equalsIgnoreCase(HeaderConfig.STREAMING_TRANSPORT)
647                         || transport.equalsIgnoreCase(SSE_TRANSPORT))) {
648                     keptOpen = true;
649                 } else if (transport != null && (
650                         transport.equalsIgnoreCase(HeaderConfig.LONG_POLLING_TRANSPORT) ||
651                                 transport.equalsIgnoreCase(HeaderConfig.JSONP_TRANSPORT))) {
652                     resumeOnBroadcast = true;
653                 }
654             }
655 
656             final Action action = (Action) request.getAttribute(NettyCometSupport.SUSPEND);
657             final State state = new State(request, action == null ? Action.CONTINUE : action);
658             ctx.getChannel().setAttachment(state);
659 
660             if (action != null && action.type() == Action.TYPE.SUSPEND) {
661                 if (action.timeout() != -1) {
662                     final AtomicReference<ChannelWriter> w = new AtomicReference<ChannelWriter>(asyncWriter);
663                     final AtomicReference<Future<?>> f = new AtomicReference<Future<?>>();
664                     f.set(suspendTimer.scheduleAtFixedRate(new Runnable() {
665                         @Override
666                         public void run() {
667                             if (!w.get().isClosed() && (System.currentTimeMillis() - w.get().lastTick()) > action.timeout()) {
668                                 AtmosphereResourceImpl impl = state.resource();
669                                 if (impl != null) {
670                                     asynchronousProcessor.endRequest(impl, false);
671                                     f.get().cancel(true);
672                                 }
673                             }
674                         }
675                     }, action.timeout(), action.timeout(), TimeUnit.MILLISECONDS));
676                 }
677             } else if (action != null && action.type() == Action.TYPE.RESUME) {
678                 resumeOnBroadcast = false;
679             }
680         } catch (AtmosphereMappingException ex) {
681             if (method.equalsIgnoreCase("GET")) {
682                 logger.trace("Unable to map the request {}, trying static file", messageEvent.getMessage());
683                 try {
684                     skipClose = true;
685                     super.messageReceived(ctx, messageEvent);
686                 } catch (Exception e) {
687                     logger.error("Unable to process request", e);
688                     throw new IOException(e);
689                 }
690             }
691         } catch (Throwable e) {
692             logger.error("Unable to process request", e);
693             throw new IOException(e);
694         } finally {
695             try {
696                 if (asyncWriter != null && !resumeOnBroadcast && !keptOpen) {
697                     if (!skipClose && response != null) {
698                         asyncWriter.close(response);
699                     } else {
700                         httpChannels.add(ctx.getChannel());
701                     }
702                 }
703             } finally {
704                 if (request != null && a != null && a.type() != Action.TYPE.SUSPEND) {
705                     request.destroy();
706                     response.destroy();
707                     framework.notify(Action.TYPE.DESTROYED, request, response);
708                 }
709             }
710         }
711     }

713     @Override
714     protected void sendError(ChannelHandlerContext ctx, HttpResponseStatus status, MessageEvent e) {
715         // For websocket, we can't send an error
716         if (websocketChannels.contains(ctx.getChannel())) {
717             logger.debug("Error {} for {}", status, e);
718             ctx.getChannel().close().addListener(ChannelFutureListener.CLOSE);
719         } else if (e != null) {
720             final HttpRequest request = (HttpRequest) e.getMessage();
721             if (HttpHeaders.getHeader(request, STATIC_MAPPING, "false").equalsIgnoreCase("false")) {
722                 super.sendError(ctx, status, e);
723             }
724         } else {
725             super.sendError(ctx, status, e);
726         }
727     }

730         isShutdown.set(true);
731         if (framework != null) framework.destroy();
732 
733         httpChannels.close();
734         websocketChannels.write(new CloseWebSocketFrame());
735         suspendTimer.shutdown();

738     @Override
739     public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent e) throws Exception {
740         super.channelClosed(ctx, e);
741         Object o = ctx.getChannel().getAttachment();
742 
743         if (o == null) return;
744 
745         if (WebSocket.class.isAssignableFrom(o.getClass())) {
746             NettyWebSocket webSocket = NettyWebSocket.class.cast(o);
747             AtmosphereResource r = webSocket.resource();
748 
749             logger.trace("Closing {}", webSocket.uuid());
750 
751             try {
752                 if (webSocket.closeFuture() != null) {
753                     webSocket.closeFuture().cancel(true);
754                 }
755 
756                 webSocketProcessor.close(webSocket, 1005);
757             } catch (Exception ex) {
758                 logger.error("{}", webSocket, ex);
759             }
760         } else if (State.class.isAssignableFrom(o.getClass())) {
761             logger.trace("State {}", o);
762             State s = State.class.cast(o);
763             if (s.action.type() == Action.TYPE.SUSPEND) {
764                 asynchronousProcessor.endRequest(s.resource(), true);
765             }
766         } else {
767             logger.error("Invalid state {} and Channel {}", o, ctx.getChannel());
768         }
769     }

771     @Override
772     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
773             throws Exception {
774         // Ignore Disconnect exception.
775         if (e.getCause() != null
776                 && (e.getCause().getClass().equals(ClosedChannelException.class)
777                 || e.getCause().getClass().equals(IOException.class))) {
778             logger.trace("Exception", e.getCause());
779         } else if (e.getCause() != null && e.getCause().getClass().equals(TooLongFrameException.class)) {
780             logger.error("TooLongFrameException. The request will be closed, make sure you increase the Config.maxChunkContentLength() to a higher value.", e.getCause());
781             super.exceptionCaught(ctx, e);
782         } else {
783             logger.debug("Exception", e.getCause());
784             super.exceptionCaught(ctx, e);
785         }
786     }

804     private void parseQueryString(Map<String, String[]> qs, String queryString) {
805         if (queryString != null) {
806             String[] s = queryString.split("&");
807             for (String a : s) {
808                 String[] q = a.split("=");
809                 String[] z = new String[]{q.length > 1 ? q[1] : ""};
810                 qs.put(q[0], z);
811             }
812         }

852     private void sendHttpResponse(ChannelHandlerContext ctx, HttpRequest req, HttpResponse res) {
853         // Generate an error page if response status code is not OK (200).
854         if (res.getStatus().getCode() != 200) {
855             res.setContent(ChannelBuffers.copiedBuffer(res.getStatus().toString(), CharsetUtil.UTF_8));
856             setContentLength(res, res.getContent().readableBytes());
857         }
858 
859         // Send the response and close the connection if necessary.
860         ChannelFuture f = ctx.getChannel().write(res);
861         if (!isKeepAlive(req) || res.getStatus().getCode() != 200) {
862             f.addListener(ChannelFutureListener.CLOSE);
863         }
864     }
