123     public BridgeRuntime(final Config config) {
124         super(config.path());
125         this.config = config;
126         framework = new AtmosphereFramework();
127 
128         if (config.broadcaster() != null) {
129             framework.setDefaultBroadcasterClassName(config.broadcaster().getName());
130         }
131 
132         framework.setAtmosphereDotXmlPath(config.configFile());
133         framework.setAsyncSupport(new NettyCometSupport(framework.getAtmosphereConfig()) {
134             @Override
135             public Action suspended(AtmosphereRequest request, AtmosphereResponse response) throws IOException, ServletException {
136                 Action a = super.suspended(request, response);
137                 if (framework.getAtmosphereConfig().isSupportSession()) {
138                     AtmosphereResource r = request.resource();
139                     HttpSession s = request.getSession(true);
140                     if (s != null) {
141                         sessions.put(r.uuid(), request.getSession(true));
142                     }
143                 }
144                 return a;
145             }
146 
147             @Override
148             public String toString() {
149                 return "NettoSphereAsyncSupport";
150             }
151         });
152 
153         try {
154             if (config.broadcasterFactory() != null) {
155                 framework.setBroadcasterFactory(config.broadcasterFactory());
156             }
157         } catch (Throwable t) {
158             logger.trace("", t);
159         }
160 
161         if (config.broadcasterCache() != null) {
162             try {
163                 framework.setBroadcasterCacheClassName(config.broadcasterCache().getName());
164             } catch (Throwable t) {
165                 logger.trace("", t);
166             }
167         }
168 
169         Map<String, AtmosphereHandler> handlersMap = config.handlersMap();
170         for (Map.Entry<String, AtmosphereHandler> e : handlersMap.entrySet()) {
171             framework.addAtmosphereHandler(e.getKey(), e.getValue());
172         }
173 
174         if (config.webSocketProtocol() != null) {
175             framework.setWebSocketProtocolClassName(config.webSocketProtocol().getName());
176         }
177 
178         for (AtmosphereInterceptor i : config.interceptors()) {
179             framework.interceptor(i);
180         }
181 
182         if (!config.scanPackages().isEmpty()) {
183             for (Class<?> s : config.scanPackages()) {
184                 framework.addAnnotationPackage(s);
185             }
186         }
187 
188         final Context context = new Context.Builder().contextPath(config.mappingPath()).basePath(config.path()).build();
189         ServletContext ctx = (ServletContext) Proxy.newProxyInstance(BridgeRuntime.class.getClassLoader(), new Class[]{ServletContext.class},
190                 new InvocationHandler() {
191                     @Override
192                     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
193                         try {
194                             Method stub = Context.class.getMethod(method.getName(), method.getParameterTypes());
195                             if (stub != null) {
196                                 return stub.invoke(context, args);
197                             } else {
198                                 logger.trace("Method {} not supported", method.getName());
199                                 return null;
200                             }
201                         } catch (NoSuchMethodException ex) {
202                             logger.trace("Method {} not supported", method.getName());
203                             return null;
204                         }
205                     }
206                 });
207 
208         try {
209             framework.externalizeDestroy(true).init(new NettyServletConfig(config.initParams(), ctx));
210         } catch (ServletException e) {
211             throw new RuntimeException(e);
212         }
213         suspendTimer = new ScheduledThreadPoolExecutor(Runtime.getRuntime().availableProcessors());
214         webSocketProcessor = WebSocketProcessorFactory.getDefault().getWebSocketProcessor(framework);
215 
216         if (config.supportChunking()) {
217             channelBufferPool = new ChannelBufferPool(0, config.writeBufferPoolSize(),
218                     config.writeBufferPoolCleanupFrequency(), framework.getAtmosphereConfig());
219         } else {
220             // Dangerous
221             channelBufferPool = null;
222         }
223 
224         for (String s : config.excludedInterceptors()) {
225             framework.excludeInterceptor(s);
226         }
227     }

233     @Override
234     public void messageReceived(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
235         Object msg = messageEvent.getMessage();
236 
237         if (isShutdown.get()) {
238             ctx.getChannel().close().addListener(ChannelFutureListener.CLOSE);
239             return;
240         }
241 
242         if (msg instanceof HttpRequest) {
243             HttpRequest r = HttpRequest.class.cast(msg);
244             // Netty fail to decode headers separated by a ','
245             List<String> c = r.getHeaders("Connection");
246             String u = r.getHeader("Upgrade");
247             boolean webSocket = false;
248             if (u != null && u.equalsIgnoreCase("websocket")) {
249                 webSocket = true;
250             }
251 
252             for (String connection : c) {
253                 if (connection != null && connection.toLowerCase().equalsIgnoreCase("upgrade")) {
254                     webSocket = true;
255                 }
256             }
257 
258             logger.trace("Handling request {}", r);
259             if (webSocket) {
260                 handleWebSocketHandshake(ctx, messageEvent);
261             } else {
262                 handleHttp(ctx, messageEvent);
263             }
264         } else if (msg instanceof WebSocketFrame) {
265             handleWebSocketFrame(ctx, messageEvent);
266         } else if (msg instanceof HttpChunk) {
267             handleHttp(ctx, messageEvent);
268         }
269     }

271     private void handleWebSocketHandshake(ChannelHandlerContext ctx, MessageEvent messageEvent) throws IOException, URISyntaxException {
272         final HttpRequest request = (HttpRequest) messageEvent.getMessage();
273 
274         // Allow only GET methods.
275         if (request.getMethod() != GET) {
276             sendHttpResponse(ctx, request, new DefaultHttpResponse(HTTP_1_1, FORBIDDEN));
277             return;
278         }
279 
280         WebSocketServerHandshakerFactory wsFactory = new WebSocketServerHandshakerFactory(getWebSocketLocation(request), null, false);
281         WebSocketServerHandshaker handshaker = wsFactory.newHandshaker(request);
282 
283         if (handshaker == null) {
284             wsFactory.sendUnsupportedWebSocketVersionResponse(ctx.getChannel());
285         } else {
286             handshaker.handshake(ctx.getChannel(), request);
287         }
288         websocketChannels.add(ctx.getChannel());
289 
290         AtmosphereRequest r = createAtmosphereRequest(ctx, request);
291         WebSocket webSocket = new NettyWebSocket(ctx.getChannel(), framework.getAtmosphereConfig());
292 
293         ctx.setAttachment(webSocket);
294         webSocketProcessor.open(webSocket, r, AtmosphereResponse.newInstance(framework.getAtmosphereConfig(), r, webSocket));
295     }

297     private void handleWebSocketFrame(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
298         WebSocketFrame frame = (WebSocketFrame) messageEvent.getMessage();
299 
300         // Check for closing frame
301         if (frame instanceof CloseWebSocketFrame) {
302             ctx.getChannel().write(frame).addListener(ChannelFutureListener.CLOSE);
303         } else if (frame instanceof PingWebSocketFrame) {
304             ctx.getChannel().write(new PongWebSocketFrame(frame.getBinaryData()));
305         } else if (frame instanceof BinaryWebSocketFrame) {
306             ChannelBuffer binaryData = frame.getBinaryData();
307             webSocketProcessor.invokeWebSocketProtocol((WebSocket) ctx.getAttachment(), binaryData.array(), binaryData.arrayOffset(), binaryData.readableBytes());
308         } else if (frame instanceof TextWebSocketFrame) {
309             webSocketProcessor.invokeWebSocketProtocol((WebSocket) ctx.getAttachment(), ((TextWebSocketFrame) frame).getText());
310         } else if (frame instanceof PongWebSocketFrame) {
311             ctx.getChannel().write(new PingWebSocketFrame(frame.getBinaryData()));
312         } else {
313             logger.warn("{} frame types not supported", frame.getClass());
314             ctx.getChannel().close();
315         }
316     }

318     private AtmosphereRequest createAtmosphereRequest(final ChannelHandlerContext ctx, final HttpRequest request) throws URISyntaxException, UnsupportedEncodingException, MalformedURLException {
319         final String base = getBaseUri(request);
320         final URI requestUri = new URI(base.substring(0, base.length() - 1) + request.getUri());
321         final String ct = HttpHeaders.getHeader(request, "Content-Type", "text/plain");
322         final long cl = HttpHeaders.getContentLength(request, 0);
323         String method = request.getMethod().getName();
324 
325         String queryString = requestUri.getQuery();
326         Map<String, String[]> qs = new HashMap<String, String[]>();
327         if (queryString != null) {
328             parseQueryString(qs, queryString);
329         }
330 
331         if (ct.equalsIgnoreCase("application/x-www-form-urlencoded")) {
332             parseQueryString(qs, new String(request.getContent().array(), "UTF-8"));
333         }
334 
335         String u = requestUri.toURL().toString();
336         int last = u.indexOf("?") == -1 ? u.length() : u.indexOf("?");
337         String url = u.substring(0, last);
338         int l;
339 
340         if (url.contains(config.mappingPath())) {
341             l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3 + config.mappingPath().length();
342         } else {
343             l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3;
344         }
345 
346         HttpSession session = null;
347         if (framework.getAtmosphereConfig().isSupportSession()) {
348             String[] transport = qs.get(HeaderConfig.X_ATMOSPHERE_TRANSPORT);
349             if (transport != null && transport.length > 0) {
350                 String[] uuid = qs.get(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
351                 if (uuid != null && uuid.length > 0) {
352                     // TODO: Session is only supported until an unsubscribe is received.
353                     if (transport[0].equalsIgnoreCase(HeaderConfig.DISCONNECT)) {
354                         sessions.remove(uuid[0]);
355                     } else {
356                         session = sessions.get(uuid[0]);
357 
358                         if (session == null) {
359                             session = new FakeHttpSession("-1", null, System.currentTimeMillis(), -1);
360                         }
361                     }
362                 }
363             }
364         }
365 
366         final Map<String, Object> attributes = new HashMap<String, Object>();
367         AtmosphereRequest.Builder requestBuilder = new AtmosphereRequest.Builder();
368         requestBuilder.requestURI(url.substring(l))
369                 .requestURL(url)
370                 .pathInfo(url.substring(l))
371                 .headers(getHeaders(request))
372                 .method(method)
373                 .contentType(ct)
374                 .contentLength(cl)
375                         // We need to read attribute after doComet
376                 .destroyable(false)
377                 .attributes(attributes)
378                 .servletPath(config.mappingPath())
379                 .session(session)
380                 .cookies(getCookies(request))
381                 .queryStrings(qs)
382                 .remoteInetSocketAddress(new Callable<InetSocketAddress>() {
383                     @Override
384                     public InetSocketAddress call() throws Exception {
385                         return (InetSocketAddress) ctx.getChannel().getRemoteAddress();
386                     }
387                 })
388                 .localInetSocketAddress(new Callable<InetSocketAddress>() {
389 
390                     @Override
391                     public InetSocketAddress call() throws Exception {
392                         return (InetSocketAddress) ctx.getChannel().getLocalAddress();
393                     }
394                 });
395 
396         ChannelBuffer internalBuffer = request.getContent();
397         if (!config.aggregateRequestBodyInMemory() && !method.equalsIgnoreCase("GET")) {
398             return requestBuilder.body(internalBuffer.array()).build();
399         } else {
400             logger.trace("Unable to read in memory the request's bytes. Using stream");
401             return requestBuilder.inputStream(new ChannelBufferInputStream(internalBuffer)).build();
402         }
403     }

405     private void handleHttp(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
406 
407         boolean skipClose = false;
408         AtmosphereResponse response = null;
409         AtmosphereRequest request = null;
410         Action a = null;
411         boolean resumeOnBroadcast = false;
412         boolean keptOpen = false;
413         ChannelWriter asyncWriter = null;
414         String method = "GET";
415         boolean writeHeader = false;
416         boolean forceSuspend = false;
417         boolean aggregateBodyInMemory = config.aggregateRequestBodyInMemory();
418 
419         try {
420             if (messageEvent.getMessage() instanceof HttpRequest) {
421                 final HttpRequest hrequest = (HttpRequest) messageEvent.getMessage();
422                 boolean ka = HttpHeaders.isKeepAlive(hrequest);
423                 asyncWriter = config.supportChunking() ?
424                         new ChunkedWriter(ctx.getChannel(), true, ka, channelBufferPool) :
425                         new StreamWriter(ctx.getChannel(), true, ka);
426 
427                 method = hrequest.getMethod().getName();
428 
429                 // First let's try to see if it's a static resources
430                 if (!hrequest.getUri().contains(HeaderConfig.X_ATMOSPHERE)) {
431                     try {
432                         hrequest.headers().add(STATIC_MAPPING, "true");
433                         super.messageReceived(ctx, messageEvent);
434 
435                         if (HttpHeaders.getHeader(hrequest, SERVICED) != null) {
436                             return;
437                         }
438                     } catch (Exception e) {
439                         logger.debug("Unexpected State", e);
440                     } finally {
441                         hrequest.headers().set(STATIC_MAPPING, "false");
442                     }
443                 }
444 
445                 request = createAtmosphereRequest(ctx, hrequest);
446                 request.setAttribute(KEEP_ALIVE, new Boolean(ka));
447 
448                 // Hacky. Is the POST doesn't contains a body, we must not close the connection yet.
449                 AtmosphereRequest.Body b = request.body();
450                 if (!aggregateBodyInMemory
451                         && !hrequest.getMethod().equals(GET)
452                         && !b.isEmpty()
453                         && (b.hasString() && b.asString().isEmpty())
454                         || (b.hasBytes() && b.asBytes().length == 0)) {
455                     forceSuspend = true;
456                 }
457             } else {
458                 request = State.class.cast(ctx.getAttachment()).request;
459                 boolean isLast = HttpChunk.class.cast(messageEvent.getMessage()).isLast();
460                 Boolean ka = (Boolean) request.getAttribute(KEEP_ALIVE);
461 
462                 asyncWriter = config.supportChunking() ?
463                         new ChunkedWriter(ctx.getChannel(), isLast, ka, channelBufferPool) :
464                         new StreamWriter(ctx.getChannel(), isLast, ka);
465                 method = request.getMethod();
466                 ChannelBuffer internalBuffer = HttpChunk.class.cast(messageEvent.getMessage()).getContent();
467 
468                 if (!aggregateBodyInMemory && internalBuffer.hasArray()) {
469                     request.body(internalBuffer.array());
470                 } else {
471                     logger.trace("Unable to read in memory the request's bytes. Using stream");
472                     request.body(new ChannelBufferInputStream(internalBuffer));
473                 }
474 
475                 if (!isLast) {
476                     forceSuspend = true;
477                 }
478             }
479 
480             response = new AtmosphereResponse.Builder()
481                     .asyncIOWriter(asyncWriter)
482                     .writeHeader(writeHeader)
483                     .destroyable(false)
484                     .header("Connection", "Keep-Alive")
485                     .header("Server", "Nettosphere/2.0")
486                     .request(request).build();
487 
488             if (config.supportChunking()) {
489                 response.setHeader("Transfer-Encoding", "chunked");
490             }
491 
492             a = framework.doCometSupport(request, response);
493             if (forceSuspend) {
494                 a.type(Action.TYPE.SUSPEND);
495                 // leave the stream open
496                 keptOpen = true;
497             }
498 
499             final AsynchronousProcessorHook hook = (AsynchronousProcessorHook) request.getAttribute(FrameworkConfig.ASYNCHRONOUS_HOOK);
500 
501             String transport = (String) request.getAttribute(FrameworkConfig.TRANSPORT_IN_USE);
502             if (transport == null) {
503                 transport = request.getHeader(X_ATMOSPHERE_TRANSPORT);
504             }
505 
506             if (a.type() == Action.TYPE.SUSPEND) {
507                 if (transport != null && (transport.equalsIgnoreCase(HeaderConfig.STREAMING_TRANSPORT)
508                         || transport.equalsIgnoreCase(SSE_TRANSPORT))) {
509                     keptOpen = true;
510                 } else if (transport != null && (
511                         transport.equalsIgnoreCase(HeaderConfig.LONG_POLLING_TRANSPORT) ||
512                                 transport.equalsIgnoreCase(HeaderConfig.JSONP_TRANSPORT))) {
513                     resumeOnBroadcast = true;
514                 }
515             }
516 
517             final Action action = (Action) request.getAttribute(NettyCometSupport.SUSPEND);
518             ctx.setAttachment(new State(hook, request));
519             if (action != null && action.type() == Action.TYPE.SUSPEND) {
520                 if (action.timeout() != -1) {
521                     final AtomicReference<ChannelWriter> w = new AtomicReference<ChannelWriter>(asyncWriter);
522                     final AtomicReference<Future<?>> f = new AtomicReference<Future<?>>();
523                     f.set(suspendTimer.scheduleAtFixedRate(new Runnable() {
524                         @Override
525                         public void run() {
526                             if (!w.get().isClosed() && (System.currentTimeMillis() - w.get().lastTick()) > action.timeout()) {
527                                 hook.timedOut();
528                                 f.get().cancel(true);
529                             }
530                         }
531                     }, action.timeout(), action.timeout(), TimeUnit.MILLISECONDS));
532                 }
533             } else if (action != null && action.type() == Action.TYPE.RESUME) {
534                 resumeOnBroadcast = false;
535             }
536         } catch (AtmosphereMappingException ex) {
537             if (method.equalsIgnoreCase("GET")) {
538                 logger.trace("Unable to map the request {}, trying static file", messageEvent.getMessage());
539                 try {
540                     skipClose = true;
541                     super.messageReceived(ctx, messageEvent);
542                 } catch (Exception e) {
543                     logger.error("Unable to process request", e);
544                     throw new IOException(e);
545                 }
546             }
547         } catch (Throwable e) {
548             logger.error("Unable to process request", e);
549             throw new IOException(e);
550         } finally {
551             try {
552                 if (asyncWriter != null && !resumeOnBroadcast && !keptOpen) {
553                     if (!skipClose && response != null) {
554                         asyncWriter.close(response);
555                     } else {
556                         httpChannels.add(ctx.getChannel());
557                     }
558                 }
559             } finally {
560                 if (request != null && a != null && a.type() != Action.TYPE.SUSPEND) {
561                     request.destroy();
562                     response.destroy();
563                     framework.notify(Action.TYPE.DESTROYED, request, response);
564                 }
565             }
566         }
567     }

569     @Override
570     protected void sendError(ChannelHandlerContext ctx, HttpResponseStatus status, MessageEvent e) {
571         // For websocket, we can't send an error
572         if (websocketChannels.contains(ctx.getChannel())) {
573             logger.debug("Error {} for {}", status, e);
574             ctx.getChannel().close().addListener(ChannelFutureListener.CLOSE);
575         } else if (e != null) {
576             final HttpRequest request = (HttpRequest) e.getMessage();
577             if (HttpHeaders.getHeader(request, STATIC_MAPPING, "false").equalsIgnoreCase("false")) {
578                 super.sendError(ctx, status, e);
579             }
580         } else {
581             super.sendError(ctx, status, e);
582         }
583     }

585     public void destroy() {
586         isShutdown.set(true);
587         httpChannels.close();
588         websocketChannels.write(new CloseWebSocketFrame());
589 
590         if (framework != null) framework.destroy();
591         suspendTimer.shutdown();
592     }

594     @Override
595     public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent e) throws Exception {
596         super.channelClosed(ctx, e);
597         Object o = ctx.getAttachment();
598 
599         if (o == null) return;
600 
601         if (WebSocket.class.isAssignableFrom(o.getClass())) {
602             WebSocket webSocket = WebSocket.class.cast(o);
603 
604             logger.trace("Closing {}", webSocket.resource().uuid());
605 
606             webSocketProcessor.close(webSocket, 1005);
607         } else if (State.class.isAssignableFrom(o.getClass())) {
608             State.class.cast(o).hook.closed();
609         }

617                 && (e.getCause().getClass().equals(ClosedChannelException.class)
618                 || e.getCause().getClass().equals(IOException.class))) {

677

691     private void sendHttpResponse(ChannelHandlerContext ctx, HttpRequest req, HttpResponse res) {
692         // Generate an error page if response status code is not OK (200).
693         if (res.getStatus().getCode() != 200) {
694             res.setContent(ChannelBuffers.copiedBuffer(res.getStatus().toString(), CharsetUtil.UTF_8));
695             setContentLength(res, res.getContent().readableBytes());
696         }
697 
698         // Send the response and close the connection if necessary.
699         ChannelFuture f = ctx.getChannel().write(res);
700         if (!isKeepAlive(req) || res.getStatus().getCode() != 200) {
701             f.addListener(ChannelFutureListener.CLOSE);
702         }
703     }
