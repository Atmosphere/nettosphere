66  /**
67   * A simple handler that serves incoming HTTP requests to send their respective
68   * HTTP responses.  It also implements {@code 'If-Modified-Since'} header to
69   * take advantage of browser cache, as described in
70   * <a href="http://tools.ietf.org/html/rfc2616#section-14.25">RFC 2616</a>.
71   *
72   * <h3>How Browser Caching Works</h3>
73   *
74   * Web browser caching works with HTTP headers as illustrated by the following
75   * sample:
76   * <ol>
77   * <li>Request #1 returns the content of <code>/file1.txt</code>.</li>
78   * <li>Contents of <code>/file1.txt</code> is cached by the browser.</li>
79   * <li>Request #2 for <code>/file1.txt</code> does return the contents of the
80   *     file again. Rather, a 304 Not Modified is returned. This tells the
81   *     browser to use the contents stored in its cache.</li>
82   * <li>The server knows the file has not been modified because the
83   *     <code>If-Modified-Since</code> date is the same as the file's last
84   *     modified date.</li>
85   * </ol>
86   *
87   * <pre>
88   * Request #1 Headers
89   * ===================
90   * GET /file1.txt HTTP/1.1
91   *
92   * Response #1 Headers
93   * ===================
94   * HTTP/1.1 200 OK
95   * Date:               Tue, 01 Mar 2011 22:44:26 GMT
96   * Last-Modified:      Wed, 30 Jun 2010 21:36:48 GMT
97   * Expires:            Tue, 01 Mar 2012 22:44:26 GMT
98   * Cache-Control:      private, max-age=31536000
99   *
100  * Request #2 Headers
101  * ===================
102  * GET /file1.txt HTTP/1.1
103  * If-Modified-Since:  Wed, 30 Jun 2010 21:36:48 GMT
104  *
105  * Response #2 Headers
106  * ===================
107  * HTTP/1.1 304 Not Modified
108  * Date:               Tue, 01 Mar 2011 22:44:28 GMT
109  *
110  * </pre>
111  */
112 public class HttpStaticFileServerHandler extends SimpleChannelUpstreamHandler {
113     public final static String STATIC_MAPPING = SimpleChannelUpstreamHandler.class.getName() + ".staticMapping";
114     public final static String SERVICED = SimpleChannelUpstreamHandler.class.getName() + ".serviced";
115 
116     private final String root;
117 
118     public HttpStaticFileServerHandler(String root) {
119         this.root = root;
120     }
121 
122     @Override
123     public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
124         HttpRequest request = (HttpRequest) e.getMessage();
125         if (request.getMethod() != GET) {
126             sendError(ctx, METHOD_NOT_ALLOWED, e);
127             return;
128         }
129 
130         String path = root + sanitizeUri(request.getUri());
131         if (path.endsWith("/") || path.endsWith(File.separator)) {
132             path += "index.html";
133         }
134 
135         if (path == null) {
136             sendError(ctx, FORBIDDEN, e);
137             return;
138         }
139 
140         File file = new File(path);
141         if (file.isHidden() || !file.exists()) {
142             sendError(ctx, NOT_FOUND, e);
143             return;
144         }
145         if (!file.isFile()) {
146             sendError(ctx, FORBIDDEN, e);
147             return;
148         }
149 
150         RandomAccessFile raf;
151         try {
152             raf = new RandomAccessFile(file, "r");
153         } catch (FileNotFoundException fnfe) {
154             sendError(ctx, NOT_FOUND, e);
155             return;
156         }
157         long fileLength = raf.length();
158 
159         HttpResponse response = new DefaultHttpResponse(HTTP_1_1, OK);
160         setContentLength(response, fileLength);
161 
162         Channel ch = e.getChannel();
163 
164         // Write the initial line and the header.
165         ch.write(response);
166 
167         // Write the content.
168         ChannelFuture writeFuture;
169         if (ch.getPipeline().get(SslHandler.class) != null) {
170             // Cannot use zero-copy with HTTPS.
171             writeFuture = ch.write(new ChunkedFile(raf, 0, fileLength, 8192));
172         } else {
173             // No encryption - use zero-copy.
174             final FileRegion region =
175                 new DefaultFileRegion(raf.getChannel(), 0, fileLength);
176             writeFuture = ch.write(region);
177             writeFuture.addListener(new ChannelFutureProgressListener() {
178                 public void operationComplete(ChannelFuture future) {
179                     region.releaseExternalResources();
180                 }
181 
182                 public void operationProgressed(
183                         ChannelFuture future, long amount, long current, long total) {
184                 }
185             });
186         }
187 
188         // Decide whether to close the connection or not.
189         if (!isKeepAlive(request)) {
190             // Close the connection when the whole content is written out.
191             writeFuture.addListener(ChannelFutureListener.CLOSE);
192         }
193         request.addHeader(SERVICED, "true");
194     }
195 
196     @Override
197     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
198             throws Exception {
199         Channel ch = e.getChannel();
200         Throwable cause = e.getCause();
201         if (cause instanceof TooLongFrameException) {
202             sendError(ctx, BAD_REQUEST, null);
203             return;
204         }
205 
206         if (ch.isConnected()) {
207             sendError(ctx, INTERNAL_SERVER_ERROR, null);
208         }
209     }
210 
211     protected String sanitizeUri(String uri) {
212         try {
213             uri = URLDecoder.decode(uri, "UTF-8");
214         } catch (UnsupportedEncodingException e) {
215             try {
216                 uri = URLDecoder.decode(uri, "ISO-8859-1");
217             } catch (UnsupportedEncodingException e1) {
218                 throw new Error();
219             }
220         }
221 
222         uri = uri.replace('/', File.separatorChar);
223 
224         if (uri.contains(File.separator + ".") ||
225                 uri.contains("." + File.separator) ||
226                 uri.startsWith(".") || uri.endsWith(".")) {
227             return null;
228         }
229 
230         return uri;
231     }
232 
233     protected void sendError(ChannelHandlerContext ctx, HttpResponseStatus status, MessageEvent e) {
234         HttpResponse response = new DefaultHttpResponse(HTTP_1_1, status);
235         response.setHeader(CONTENT_TYPE, "text/plain; charset=UTF-8");
236         response.setHeader(CONTENT_LENGTH, "0");
237         response.setHeader("Server", "Atmosphere-" + Version.getRawVersion());
238 
239         // Close the connection as soon as the error message is sent.
240         ctx.getChannel().write(response).addListener(ChannelFutureListener.CLOSE);
241     }
