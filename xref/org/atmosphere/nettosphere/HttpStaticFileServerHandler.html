85  /**
86   * A simple handler that serves incoming HTTP requests to send their respective
87   * HTTP responses.  It also implements {@code 'If-Modified-Since'} header to
88   * take advantage of browser cache, as described in
89   * <a href="http://tools.ietf.org/html/rfc2616#section-14.25">RFC 2616</a>.
90   * <p/>
91   * <h3>How Browser Caching Works</h3>
92   * <p/>
93   * Web browser caching works with HTTP headers as illustrated by the following
94   * sample:
95   * <ol>
96   * <li>Request #1 returns the content of <code>/file1.txt</code>.</li>
97   * <li>Contents of <code>/file1.txt</code> is cached by the browser.</li>
98   * <li>Request #2 for <code>/file1.txt</code> does return the contents of the
99   * file again. Rather, a 304 Not Modified is returned. This tells the
100  * browser to use the contents stored in its cache.</li>
101  * <li>The server knows the file has not been modified because the
102  * <code>If-Modified-Since</code> date is the same as the file's last
103  * modified date.</li>
104  * </ol>
105  * <p/>
106  * <pre>
107  * Request #1 Headers
108  * ===================
109  * GET /file1.txt HTTP/1.1
110  *
111  * Response #1 Headers
112  * ===================
113  * HTTP/1.1 200 OK
114  * Date:               Tue, 01 Mar 2011 22:44:26 GMT
115  * Last-Modified:      Wed, 30 Jun 2010 21:36:48 GMT
116  * Expires:            Tue, 01 Mar 2013 22:44:26 GMT
117  * Cache-Control:      private, max-age=31536000
118  *
119  * Request #2 Headers
120  * ===================
121  * GET /file1.txt HTTP/1.1
122  * If-Modified-Since:  Wed, 30 Jun 2010 21:36:48 GMT
123  *
124  * Response #2 Headers
125  * ===================
126  * HTTP/1.1 304 Not Modified
127  * Date:               Tue, 01 Mar 2011 22:44:28 GMT
128  *
129  * </pre>
130  */
131 public class HttpStaticFileServerHandler extends SimpleChannelUpstreamHandler {
132     public static final String HTTP_DATE_FORMAT = "EEE, dd MMM yyyy HH:mm:ss zzz";
133     public static final String HTTP_DATE_GMT_TIMEZONE = "GMT";
134     public static final int HTTP_CACHE_SECONDS = 60;
135 
136     private final static Logger logger = LoggerFactory.getLogger(HttpStaticFileServerHandler.class);
137 
138     public final static String STATIC_MAPPING = SimpleChannelUpstreamHandler.class.getName() + ".staticMapping";
139     public final static String SERVICED = SimpleChannelUpstreamHandler.class.getName() + ".serviced";
140     private final List<String> paths;
141     private String defaultContentType = "text/html";
142 
143     public HttpStaticFileServerHandler(List<String> paths) {
144         this.paths = paths;
145     }
146 
147     @Override
148     public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
149         HttpRequest request = (HttpRequest) e.getMessage();
150         RandomAccessFile raf = null;
151         boolean found = true;
152         File file = null;
153         for (String p : paths) {
154             String path = p + sanitizeUri(request.getUri());
155             if (path.endsWith("/") || path.endsWith(File.separator)) {
156                 path += "index.html";
157             }
158 
159             if (path == null) {
160                 found = false;
161                 continue;
162             }
163 
164             file = new File(path);
165             if (file.isHidden() || !file.exists()) {
166                 found = false;
167                 continue;
168             }
169             if (!file.isFile()) {
170                 found = false;
171                 continue;
172             }
173 
174             try {
175                 raf = new RandomAccessFile(file, "r");
176                 found = true;
177                 break;
178             } catch (FileNotFoundException fnfe) {
179                 found = false;
180                 continue;
181             }
182         }
183 
184         if (!found) {
185             sendError(ctx, NOT_FOUND, e);
186             return;
187         }
188 
189         // Cache Validation
190         String ifModifiedSince = request.getHeader(IF_MODIFIED_SINCE);
191         if (file != null && ifModifiedSince != null && ifModifiedSince.length() != 0) {
192             SimpleDateFormat dateFormatter = new SimpleDateFormat(HTTP_DATE_FORMAT, Locale.US);
193             Date ifModifiedSinceDate = dateFormatter.parse(ifModifiedSince);
194 
195             // Only compare up to the second because the datetime format we send to the client does
196             // not have milliseconds
197             long ifModifiedSinceDateSeconds = ifModifiedSinceDate.getTime() / 1000;
198             long fileLastModifiedSeconds = file.lastModified() / 1000;
199             if (ifModifiedSinceDateSeconds == fileLastModifiedSeconds) {
200                 sendNotModified(ctx);
201                 return;
202             }
203         }
204 
205         long fileLength = raf.length();
206 
207         HttpResponse response = new DefaultHttpResponse(HTTP_1_1, OK);
208         contentType(request, response, file);
209         setContentLength(response, fileLength);
210         setDateAndCacheHeaders(response,file);
211 
212         Channel ch = e.getChannel();
213 
214         // Write the initial line and the header.
215         ch.write(response);
216 
217         // Write the content.
218         ChannelFuture writeFuture;
219         if (ch.getPipeline().get(SslHandler.class) != null) {
220             // Cannot use zero-copy with HTTPS.
221             writeFuture = ch.write(new ChunkedFile(raf, 0, fileLength, 8192));
222         } else {
223             // No encryption - use zero-copy.
224             final FileRegion region =
225                     new DefaultFileRegion(raf.getChannel(), 0, fileLength);
226             writeFuture = ch.write(region);
227             writeFuture.addListener(new ChannelFutureProgressListener() {
228                 public void operationComplete(ChannelFuture future) {
229                     region.releaseExternalResources();
230                 }
231 
232                 public void operationProgressed(
233                         ChannelFuture future, long amount, long current, long total) {
234                 }
235             });
236         }
237 
238         // Decide whether to close the connection or not.
239         if (!isKeepAlive(request)) {
240             // Close the connection when the whole content is written out.
241             writeFuture.addListener(ChannelFutureListener.CLOSE);
242         }
243         request.addHeader(SERVICED, "true");
244     }

246     @Override
247     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
248             throws Exception {
249         Channel ch = e.getChannel();
250 
251         // Prevent recursion when the client close the connection during a write operation. In that
252         // scenario the sendError will be invoked, but will fail since the channel has already been closed
253         // For an unknown reason,
254         if (ch.getAttachment() != null && Error.class.isAssignableFrom(ch.getAttachment().getClass())) {
255             return;
256         }
257 
258         Throwable cause = e.getCause();
259         if (cause instanceof TooLongFrameException) {
260             sendError(ctx, BAD_REQUEST, null);
261             return;
262         }
263 
264         ch.setAttachment(new Error());
265         if (ch.isOpen()) {
266             sendError(ctx, INTERNAL_SERVER_ERROR, null);
267         }
268     }

270     protected String sanitizeUri(String uri) {
271         try {
272             uri = URLDecoder.decode(uri, "UTF-8");
273         } catch (UnsupportedEncodingException e) {
274             try {
275                 uri = URLDecoder.decode(uri, "ISO-8859-1");
276             } catch (UnsupportedEncodingException e1) {
277                 throw new Error();
278             }
279         }
280 
281         uri = uri.replace('/', File.separatorChar);
282 
283         if (uri.contains(File.separator + ".") ||
284                 uri.contains("." + File.separator) ||
285                 uri.startsWith(".") || uri.endsWith(".")) {
286             return null;
287         }
288 
289         int pos = uri.indexOf("?");
290         if (pos != -1) {
291             uri = uri.substring(0, pos);
292         }
293         return uri;
294     }

296     private static void sendNotModified(ChannelHandlerContext ctx) {
297         HttpResponse response = new DefaultHttpResponse(HTTP_1_1, NOT_MODIFIED);
298         setDateHeader(response);
299 
300         // Close the connection as soon as the error message is sent.
301         ctx.getChannel().write(response).addListener(ChannelFutureListener.CLOSE);
302     }
303 
304     private static void setDateHeader(HttpResponse response) {
305         SimpleDateFormat dateFormatter = new SimpleDateFormat(HTTP_DATE_FORMAT, Locale.US);
306         dateFormatter.setTimeZone(TimeZone.getTimeZone(HTTP_DATE_GMT_TIMEZONE));
307 
308         Calendar time = new GregorianCalendar();
309         response.setHeader(DATE, dateFormatter.format(time.getTime()));
310     }
311 
312     protected void sendError(ChannelHandlerContext ctx, HttpResponseStatus status, MessageEvent event) {
313         logger.trace("Error {} for {}", status, event);
314         HttpResponse response = new DefaultHttpResponse(HTTP_1_1, status);
315         response.setHeader(CONTENT_TYPE, "text/plain; charset=UTF-8");
316         response.setHeader(CONTENT_LENGTH, "0");
317         response.setHeader("Server", "Atmosphere-" + Version.getRawVersion());
318 
319         // Close the connection as soon as the error message is sent.
320         ctx.getChannel().write(response).addListener(ChannelFutureListener.CLOSE);
321     }
322 
323     protected void contentType(HttpRequest request, HttpResponse response, File resource) {
324         String substr;
325         String uri = request.getUri();
326         int dot = uri.lastIndexOf(".");
327         if (dot < 0) {
328             substr = resource.toString();
329             dot = substr.lastIndexOf(".");
330         } else {
331             substr = uri;
332         }
333 
334         if (dot > 0) {
335             String ext = substr.substring(dot + 1);
336             String contentType = MimeType.get(ext, defaultContentType);
337             response.addHeader(HttpHeaders.Names.CONTENT_TYPE, contentType);
338         } else {
339             response.addHeader(HttpHeaders.Names.CONTENT_TYPE, defaultContentType);
340         }
341 
342     }

344     private static void setDateAndCacheHeaders(HttpResponse response, File fileToCache) {
345         SimpleDateFormat dateFormatter = new SimpleDateFormat(HTTP_DATE_FORMAT, Locale.US);
346         dateFormatter.setTimeZone(TimeZone.getTimeZone(HTTP_DATE_GMT_TIMEZONE));
347 
348         // Date header
349         Calendar time = new GregorianCalendar();
350         response.setHeader(DATE, dateFormatter.format(time.getTime()));
351 
352         // Add cache headers
353         time.add(Calendar.SECOND, HTTP_CACHE_SECONDS);
354         response.setHeader(EXPIRES, dateFormatter.format(time.getTime()));
355         response.setHeader(CACHE_CONTROL, "private, max-age=" + HTTP_CACHE_SECONDS);
356         response.setHeader(
357                 LAST_MODIFIED, dateFormatter.format(new Date(fileToCache.lastModified())));
358     }
