16  package org.atmosphere.nettosphere;
17  
18  import org.jboss.netty.bootstrap.ServerBootstrap;
19  import org.jboss.netty.channel.Channel;
20  import org.jboss.netty.channel.ChannelPipelineFactory;
21  import org.jboss.netty.channel.group.ChannelGroup;
22  import org.jboss.netty.channel.group.ChannelGroupFuture;
23  import org.jboss.netty.channel.group.DefaultChannelGroup;
24  import org.jboss.netty.channel.socket.nio.NioServerSocketChannelFactory;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import java.io.BufferedReader;
29  import java.io.InputStreamReader;
30  import java.net.InetSocketAddress;
31  import java.net.SocketAddress;
32  import java.util.concurrent.Executors;
33  
34  /**
35   * Start Atmosphere on top of Netty. To configure Atmosphere, use the {@link Config}.
36   */
37  public final class Nettosphere {
38  
39      private static final Logger logger = LoggerFactory.getLogger(Nettosphere.class);
40      private static final ChannelGroup ALL_CHANNELS = new DefaultChannelGroup("atmosphere");
41      private final ChannelPipelineFactory pipelineFactory;
42      private final ServerBootstrap bootstrap;
43      private final SocketAddress localSocket;
44      private final NettyAtmosphereHandler handler;
45  
46      private Nettosphere(Config config) {
47          handler = new NettyAtmosphereHandler(config);
48          this.pipelineFactory = new AtmosphereChannelPipelineFactory(handler);
49          this.localSocket = new InetSocketAddress(config.host(), config.port());
50          this.bootstrap = buildBootstrap();
51      }
52  
53      /**
54       * Start the server
55       */
56      public void start() {
57          final Channel serverChannel = bootstrap.bind(localSocket);
58          ALL_CHANNELS.add(serverChannel);
59      }
60  
61      /**
62       * Stop the Server
63       */
64      public void stop() {
65          handler.destroy();
66          final ChannelGroupFuture future = ALL_CHANNELS.close();
67          future.awaitUninterruptibly();
68          bootstrap.getFactory().releaseExternalResources();
69          ALL_CHANNELS.clear();
70      }
71  
72      private ServerBootstrap buildBootstrap() {
73          final ServerBootstrap bootstrap = new ServerBootstrap(
74                  new NioServerSocketChannelFactory(
75                          Executors.newCachedThreadPool(),
76                          Executors.newCachedThreadPool()));
77  
78          bootstrap.setPipelineFactory(pipelineFactory);
79          return bootstrap;
80      }
81  
82      /**
83       * Construct a {@link Nettosphere}.
84       */
85      public final static class Builder {
86  
87          private Config config = new Config.Builder().build();
88  
89          public Builder config(Config config) {
90              this.config = config;
91              return this;
92          }
93  
94          public Nettosphere build() {
95              return new Nettosphere(config);
96          }
97  
98      }
99  
100     public static void main(String[] args) throws Exception {
101         Config.Builder b = new Config.Builder();
102         b.resource(args[0]).port(8080).host("127.0.0.1");
103         Nettosphere s = new Nettosphere(b.build());
104         s.start();
105         String a = "";
106 
107         logger.info("NettoSphere Server started");
108         logger.info("Type quit to stop the server");
109         BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
110         while (!(a.equals("quit"))) {
111             a = br.readLine();
112         }
113         System.exit(-1);
114     }
