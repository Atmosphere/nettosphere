37  /**
38   * Start Atmosphere on top of Netty. To configure Atmosphere, use the {@link Config}.  As simple as
39   * <blockquote><pre>
40   * <p/>
41   * Config config = new Config.Builder()
42   *  .port(port)
43   *  .host("127.0.0.1")
44   *  .initParam("foo", "bar")
45   *  .resource("/", new AtmosphereHandlerAdapter() {
46   *   <p/>
47   *    public void onStateChange(AtmosphereResourceEvent r) throws IOException {
48   *    }
49   *   <p/>
50   * }).build();
51   * <p/>
52   * server = new Nettosphere.Builder().config(config).build();
53   * </pre></blockquote>
54   *
55   * @author Jeanfrancois Arcand
56   */
57  public final class Nettosphere {
58  
59      public final static String FLASH_SUPPORT = Nettosphere.class.getName() + ".enableFlash";
60      private static final Logger logger = LoggerFactory.getLogger(Nettosphere.class);
61      private static final ChannelGroup ALL_CHANNELS = new DefaultChannelGroup("atmosphere");
62      private final ChannelPipelineFactory pipelineFactory;
63      private final ServerBootstrap bootstrap;
64      private final SocketAddress localSocket;
65      private final BridgeRuntime handler;
66      private final AtomicBoolean started = new AtomicBoolean();
67      private final ServerBootstrap bootstrapFlashPolicy;
68      private final SocketAddress localPolicySocket;
69  
70      private Nettosphere(Config config) {
71          handler = new BridgeRuntime(config);
72          this.pipelineFactory = new NettyPipelineFactory(handler);
73          this.localSocket = new InetSocketAddress(config.host(), config.port());
74          this.bootstrap = buildBootstrap();
75  
76          configureBootstrap(bootstrap, config);
77  
78          if (config.initParams().containsKey(FLASH_SUPPORT)) {
79              this.bootstrapFlashPolicy = buildBootstrapFlashPolicy();
80              localPolicySocket = new InetSocketAddress(843);
81          } else {
82              this.bootstrapFlashPolicy = null;
83              localPolicySocket = null;
84          }
85      }
86  
87      private void configureBootstrap(ServerBootstrap bootstrap, Config config) {
88          bootstrap.setOption("child.tcpNoDelay", config.socketNoTcpDelay());
89          bootstrap.setOption("child.keepAlive", config.socketKeepAlive());
90      }
91  
92      /**
93       * Return the {@link org.atmosphere.cpr.AtmosphereFramework} instance
94       * @return the {@link AtmosphereFramework} instance
95       */
96      public AtmosphereFramework framework() {
97          return handler.framework();
98      }
99  
100     /**
101      * Start the server
102      */
103     public void start() {
104         final Channel serverChannel = bootstrap.bind(localSocket);
105         ALL_CHANNELS.add(serverChannel);
106         started.set(true);
107         if (bootstrapFlashPolicy != null) {
108             bootstrapFlashPolicy.bind(localPolicySocket);
109         }
110     }
111 
112     /**
113      * Stop the Server
114      */
115     public void stop() {
116         try {
117             handler.destroy();
118             final ChannelGroupFuture future = ALL_CHANNELS.close();
119             future.awaitUninterruptibly();
120             bootstrap.getFactory().releaseExternalResources();
121             ALL_CHANNELS.clear();
122         } finally {
123             started.set(false);
124         }
125     }
126 
127     /**
128      * Return true is the server was successfully started.
129      *
130      * @return true is the server was successfully started.
131      */
132     public boolean isStarted() {
133         return started.get();
134     }
135 
136     private ServerBootstrap buildBootstrap() {
137         final ServerBootstrap bootstrap = new ServerBootstrap(
138                 new NioServerSocketChannelFactory(
139                         Executors.newCachedThreadPool(),
140                         Executors.newCachedThreadPool()));
141 
142         bootstrap.setPipelineFactory(pipelineFactory);
143         return bootstrap;
144     }
145 
146     private ServerBootstrap buildBootstrapFlashPolicy() {
147         // Configure the server.
148         final ServerBootstrap bootstrap = new ServerBootstrap(
149                 new NioServerSocketChannelFactory(
150                         Executors.newCachedThreadPool(),
151                         Executors.newCachedThreadPool()));
152 
153         // Set up the event pipeline factory.
154         bootstrap.setPipelineFactory(new FlashPolicyServerPipelineFactory());
155         return bootstrap;
156     }
157 
158     /**
159      * Construct a {@link Nettosphere}.
160      */
161     public final static class Builder {
162 
163         private Config config = new Config.Builder().build();
164 
165         public Builder config(Config config) {
166             this.config = config;
167             return this;
168         }
169 
170         public Nettosphere build() {
171             return new Nettosphere(config);
172         }
173 
174     }
175 
176     public static void main(String[] args) throws Exception {
177         Config.Builder b = new Config.Builder();
178         b.resource(args[0]).port(8080).host("127.0.0.1");
179         Nettosphere s = new Nettosphere(b.build());
180         s.start();
181         String a = "";
182 
183         logger.info("NettoSphere Server started");
184         logger.info("Type quit to stop the server");
185         BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
186         while (!(a.equals("quit"))) {
187             a = br.readLine();
188         }
189         System.exit(-1);
190     }
