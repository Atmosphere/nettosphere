272         /**
273          * Add an Jersey/JAX RS that will be mapped to the specified path
274          *
275          * @param path a mapping path
276          * @param c    an Jersey/JAX RS
277          * @return this
278          */
279         public Builder resource(String path, Class<?> c) {
280             try {
281                 if (AtmosphereHandler.class.isAssignableFrom(c)) {
282                     handlers.put(path, AtmosphereHandler.class.cast(c.newInstance()));
283                 } else if (Servlet.class.isAssignableFrom(c)) {
284                     handlers.put(path, new ReflectorServletProcessor(Servlet.class.cast(c.newInstance())));
285                 } else {
286                     // TODO: NOT clear
287                     staticResourcePath = path;
288                     initParam("com.sun.jersey.config.property.packages", c.getPackage().getName());
289                 }
290             } catch (Exception ex) {
291                 logger.error("Invalid resource {}", c);
292             }
293             return this;
294         }
