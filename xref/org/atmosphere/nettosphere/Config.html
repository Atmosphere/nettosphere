361         /**
362          * Add an {@link AtmosphereHandler} or {@link Servlet} class
363          *
364          * @param path a mapping path
365          * @param c    an {@link AtmosphereHandler} or {@link Servlet} class
366          * @return this
367          */
368         public Builder resource(String path, Class<?> c) {
369             try {
370                 if (AtmosphereHandler.class.isAssignableFrom(c)) {
371                     handlers.put(path, AtmosphereHandler.class.cast(c.newInstance()));
372                 } else if (Servlet.class.isAssignableFrom(c)) {
373                     handlers.put(path, new ReflectorServletProcessor(Servlet.class.cast(c.newInstance())));
374                 } else {
375                     throw new IllegalStateException("You class must implements AtmosphereHandler or be a Servlet");
376                 }
377             } catch (Exception ex) {
378                 logger.error("Invalid resource {}", c);
379             }
380             return this;
381         }

475         /**
476          * By default, Nettosphere aggregate the HTTP request's body in memory an invoke an Atmosphere's components with
477          * a single {@link AtmosphereResource}. Setting supportChunkAggregator to false will instead invoke Atmosphere's component
478          * with a new {@link AtmosphereResource} each time the request's body is read in memory. Setting to false
479          * may significantly increase the performance and reduce memory footprint. Note that setting this value to false
480          * may deliver to your Atmosphere's component partial body, so your application must make sure to aggregate the
481          * body before parsing the data if needed. For example, if you are using JSON as format, make sure you parse the
