193          * @return this
194          */
195         public Builder handler(String path, final Handler handler) {
196             handlers.put(path, new AbstractReflectorAtmosphereHandler() {
197                 @Override
198                 public void onRequest(AtmosphereResource resource) throws IOException {
199                     handler.handle(resource);
200                 }

202                 @Override
203                 public void destroy() {
204                 }
205             });
206             return this;
207         }

209         /**

218         /**
219          * Add an Jersey/JAX RS that will be mapped to the specified path
220          *
221          * @param path a mapping path
222          * @param c    an Jersey/JAX RS
223          * @return this
224          */
225         public Builder resource(String path, Class<?> c) {
226             try {
227                 if (AtmosphereHandler.class.isAssignableFrom(c)) {
228                     handlers.put(path, AtmosphereHandler.class.cast(c.newInstance()));
229                 } else if (Servlet.class.isAssignableFrom(c)) {
230                     handlers.put(path, new ReflectorServletProcessor(Servlet.class.cast(c.newInstance())));
231                 } else {
232                     // TODO: NOT clear
233                     staticResourcePath = path;
234                     initParam("com.sun.jersey.config.property.packages", c.getPackage().getName());
235                 }
236             } catch (Exception ex) {
237                 logger.error("Invalid resource {}", c);
238             }
239             return this;
240         }
