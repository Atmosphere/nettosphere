140         public Builder resource(String path, Class<?> c)  {
141             try {
142                 if (AtmosphereHandler.class.isAssignableFrom(c)) {
143                     handlers.put(path, AtmosphereHandler.class.cast(c.newInstance()));
144                 } else if (Servlet.class.isAssignableFrom(c)) {
145                     handlers.put(path, new ReflectorServletProcessor(Servlet.class.cast(c.newInstance())));
146                 } else {
147                     // TODO: NOT clear
148                     staticResourcePath = path;
149                     initParam("com.sun.jersey.config.property.packages", c.getPackage().getName());
150                 }
151             } catch (Exception ex) {
152                 logger.error("Invalid resource {}", c);
153             }
154             return this;
155         }
