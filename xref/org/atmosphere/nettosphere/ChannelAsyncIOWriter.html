32  import java.util.concurrent.atomic.AtomicBoolean;

35  /**
36   * An {@link AsyncIOWriter} that bridge Atmosphere output stream with Netty's Channel.
37   */
38  public class ChannelAsyncIOWriter implements AsyncIOWriter {
39      private static final Logger logger = LoggerFactory.getLogger(ChannelAsyncIOWriter.class);
40  
41      private final Channel channel;
42      private final AtomicInteger pendingWrite = new AtomicInteger();
43      private final AtomicBoolean asyncClose = new AtomicBoolean(false);
44      private final ML listener = new ML();
45      private boolean resumeOnBroadcast = false;
46      private boolean byteWritten = false;
47      private final AtomicBoolean isClosed = new AtomicBoolean(false);
48      private boolean headerWritten = false;
49      private final static String END = Integer.toHexString(0);
50      private final static byte[] CHUNK_DELIMITER = "\r\n".getBytes();
51      private final static byte[] ENDCHUNK = (END + "\r\n\r\n").getBytes();
52      private long lastWrite = 0;
53  
54      public ChannelAsyncIOWriter(Channel channel) {
55          this.channel = channel;
56      }
57  
58      public boolean isClosed() {
59          return isClosed.get();
60      }
61  
62      public boolean byteWritten() {
63          return byteWritten;
64      }
65  
66      public void resumeOnBroadcast(boolean resumeOnBroadcast) {
67          this.resumeOnBroadcast = resumeOnBroadcast;
68      }
69  
70      @Override
71      public void redirect(String location) throws IOException {
72          throw new UnsupportedOperationException();
73      }
74  
75      @Override
76      public void writeError(int errorCode, String message) throws IOException {
77          if (!channel.isOpen()) {
78              return;
79          }
80  
81          try {
82              DefaultHttpResponse response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
83                      HttpResponseStatus.valueOf(errorCode));
84              channel.write(response).addListener(ChannelFutureListener.CLOSE);
85          } catch (Throwable ex) {
86              logger.debug("", ex);
87          }
88      }
89  
90      @Override
91      public void write(String data) throws IOException {
92          byte[] b = data.getBytes("ISO-8859-1");
93          write(b);
94      }
95  
96      @Override
97      public void write(byte[] data) throws IOException {
98          write(data, 0, data.length);
99      }
100 
101     @Override
102     public void write(byte[] data, int offset, int length) throws IOException {
103 
104         if (channel.isOpen()) {
105             pendingWrite.incrementAndGet();
106             final ChannelBuffer buffer = ChannelBuffers.dynamicBuffer();
107 
108             ChannelBufferOutputStream c = new ChannelBufferOutputStream(buffer);
109 
110             if (headerWritten) {
111                 c.write(Integer.toHexString(length - offset).getBytes("UTF-8"));
112                 c.write(CHUNK_DELIMITER);
113             }
114 
115             c.write(data, offset, length);
116             if (headerWritten) {
117                 c.write(CHUNK_DELIMITER);
118             }
119 
120             channel.write(c.buffer()).addListener(listener);
121             byteWritten = true;
122             lastWrite = System.currentTimeMillis();
123         } else {
124             logger.warn("Trying to write on a closed channel {}", channel);
125         }
126         headerWritten = true;
127     }
128 
129     public long lastTick() {
130         return lastWrite == -1 ? System.currentTimeMillis() : lastWrite;
131     }
132 
133     @Override
134     public void flush() {
135     }
136 
137     @Override
138     public void close() throws IOException {
139         asyncClose.set(true);
140         if (pendingWrite.get() == 0 && channel.isOpen()) {
141             _close();
142         }
143     }
144 
145     private final class ML implements ChannelFutureListener {
146         @Override
147         public void operationComplete(ChannelFuture future) throws Exception {
148             if (channel.isOpen() && (!future.isSuccess() || (pendingWrite.decrementAndGet() == 0 && (resumeOnBroadcast || asyncClose.get())))) {
149                 _close();
150             }
151         }
152     }
153 
154     void _close() {
155         if (!isClosed.getAndSet(true)) {
156             headerWritten = false;
157             final ChannelBuffer buffer = ChannelBuffers.dynamicBuffer();
158             ChannelBufferOutputStream c = new ChannelBufferOutputStream(buffer);
159             try {
160                 c.write(ENDCHUNK);
161                 channel.write(buffer).addListener(ChannelFutureListener.CLOSE);
162             } catch (IOException e) {
163                 logger.trace("Close error", e);
164             }
165         }
166     }
