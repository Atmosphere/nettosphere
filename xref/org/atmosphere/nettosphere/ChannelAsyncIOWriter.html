39  /**
40   * An {@link AsyncIOWriter} that bridge Atmosphere output stream with Netty's Channel.
41   */
42  public class ChannelAsyncIOWriter extends AtmosphereInterceptorWriter {
43      private static final Logger logger = LoggerFactory.getLogger(ChannelAsyncIOWriter.class);
44  
45      private final Channel channel;
46      private final AtomicInteger pendingWrite = new AtomicInteger();
47      private final AtomicBoolean asyncClose = new AtomicBoolean(false);
48      private final ML listener = new ML();
49      private boolean resumeOnBroadcast = false;
50      private boolean byteWritten = false;
51      private final AtomicBoolean isClosed = new AtomicBoolean(false);
52      private boolean headerWritten = false;
53      private final static String END = Integer.toHexString(0);
54      private final static byte[] CHUNK_DELIMITER = "\r\n".getBytes();
55      private final static byte[] ENDCHUNK = (END + "\r\n\r\n").getBytes();
56      private long lastWrite = 0;
57      private final ByteArrayAsyncWriter buffer = new ByteArrayAsyncWriter();
58      private final boolean writeHeader;

82      @Override
83      public AsyncIOWriter writeError(AtmosphereResponse r, int errorCode, String message) throws IOException {
84          if (!channel.isOpen()) {
85              return this;
86          }
87  
88          try {
89              DefaultHttpResponse response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
90                      HttpResponseStatus.valueOf(errorCode));
91              channel.write(response).addListener(ChannelFutureListener.CLOSE);
92          } catch (Throwable ex) {
93              logger.debug("", ex);
94          }
95          return this;
96      }

111     protected byte[] transform(AtmosphereResponse response, byte[] b, int offset, int length) throws IOException {
112         AsyncIOWriter a = response.getAsyncIOWriter();
113         try {
114             response.asyncIOWriter(buffer);
115             invokeInterceptor(response, b, offset, length);
116             return buffer.stream().toByteArray();
117         } finally {
118             buffer.close(null);
119             response.asyncIOWriter(a);
120         }
121     }

123     @Override
124     public AsyncIOWriter write(AtmosphereResponse r, byte[] data, int offset, int length) throws IOException {
125 
126         boolean transform = filters.size() > 0 && r.getStatus() < 400;
127         if (transform) {
128             data = transform(r, data, offset, length);
129             offset = 0;
130             length = data.length;
131         }
132 
133         logger.trace("About to write to {}", r.resource() != null ? r.resource().uuid() : "null");
134         if (channel.isOpen()) {
135             final ChannelBuffer buffer = ChannelBuffers.dynamicBuffer();
136             pendingWrite.incrementAndGet();
137             if (writeHeader && !headerWritten) {
138                 buffer.writeBytes(constructStatusAndHeaders(r).getBytes("UTF-8"));
139                 headerWritten = true;
140             }
141 
142             ChannelBufferOutputStream c = new ChannelBufferOutputStream(buffer);
143 
144             if (headerWritten) {
145                 c.write(Integer.toHexString(length - offset).getBytes("UTF-8"));
146                 c.write(CHUNK_DELIMITER);
147             }
148 
149             c.write(data, offset, length);
150             if (headerWritten) {
151                 c.write(CHUNK_DELIMITER);
152             }
153 
154             channel.write(c.buffer()).addListener(listener);
155             byteWritten = true;
156             lastWrite = System.currentTimeMillis();
157         } else {
158             logger.debug("Trying to write on a closed channel {}", channel);
159             throw new IOException("Channel closed");
160         }
161         headerWritten = true;
162         return this;
163     }

165     public long lastTick() {
166         return lastWrite == -1 ? System.currentTimeMillis() : lastWrite;
167     }

169     @Override
170     public void close(AtmosphereResponse r) throws IOException {
171 
172         if (!channel.isOpen()) return;
173 
174         if (writeHeader && !headerWritten && r != null) {
175             final ChannelBuffer buffer = ChannelBuffers.dynamicBuffer();
176 
177             buffer.writeBytes(constructStatusAndHeaders(r).getBytes("UTF-8"));
178             channel.write(buffer);
179             headerWritten = true;
180         }
181 
182         // Make sure we don't have bufferred bytes
183         if (!byteWritten && r != null && r.getOutputStream() != null) {
184             r.getOutputStream().flush();
185         }
186 
187         asyncClose.set(true);
188         if (pendingWrite.get() == 0 && channel.isOpen()) {
189             _close();
190         }
191     }
192 
193     private final class ML implements ChannelFutureListener {
194         @Override
195         public void operationComplete(ChannelFuture future) throws Exception {
196             if (channel.isOpen() && (!future.isSuccess() || (pendingWrite.decrementAndGet() == 0 && (resumeOnBroadcast || asyncClose.get())))) {
197                 _close();
198             }
199         }
200     }
201 
202     void _close() {
203         if (!isClosed.getAndSet(true)) {
204             headerWritten = false;
205             final ChannelBuffer buffer = ChannelBuffers.dynamicBuffer();
206             ChannelBufferOutputStream c = new ChannelBufferOutputStream(buffer);
207             try {
208                 c.write(ENDCHUNK);
209                 channel.write(buffer).addListener(ChannelFutureListener.CLOSE);
210             } catch (IOException e) {
211                 logger.trace("Close error", e);
212             }
213         }
214     }

216     // Duplicate from AtmosphereResource.constructStatusAndHeaders
217     private String constructStatusAndHeaders(AtmosphereResponse response) {
218         StringBuffer b = new StringBuffer("HTTP/1.1")
219                 .append(" ")
220                 .append(response.getStatus())
221                 .append(" ")
222                 .append(response.getStatusMessage())
223                 .append("\n");
224 
225         Map<String,String> headers = response.headers();
226         String contentType = response.getContentType();
227         int contentLength = -1; //FIX ME
228 
229         b.append("Content-Type").append(":").append(headers.get("Content-Type") == null ? contentType : headers.get("Content-Type")).append("\n");
230         if (contentLength != -1) {
231             b.append("Content-Length").append(":").append(contentLength).append("\n");
232         }
233 
234         for (String s : headers.keySet()) {
235             if (!s.equalsIgnoreCase("Content-Type")) {
236                 b.append(s).append(":").append(headers.get(s)).append("\n");
237             }
238         }
239         b.deleteCharAt(b.length() - 1);
240         b.append("\r\n\r\n");
241         return b.toString();
242     }
