16  package org.atmosphere.nettosphere;
17  
18  import org.jboss.netty.channel.ChannelPipeline;
19  import org.jboss.netty.channel.ChannelPipelineFactory;
20  import org.jboss.netty.channel.ChannelUpstreamHandler;
21  import org.jboss.netty.handler.codec.http.HttpChunkAggregator;
22  import org.jboss.netty.handler.codec.http.HttpRequestDecoder;
23  import org.jboss.netty.handler.codec.http.HttpResponseEncoder;
24  import org.jboss.netty.handler.ssl.SslHandler;
25  import org.jboss.netty.handler.stream.ChunkedWriteHandler;
26  
27  import javax.net.ssl.SSLEngine;
28  
29  import static org.jboss.netty.channel.Channels.pipeline;
30  
31  class NettyPipelineFactory implements
32          ChannelPipelineFactory {
33  
34      private final transient BridgeRuntime bridgeRuntime;
35      private final transient Config config;
36  
37      public NettyPipelineFactory(final BridgeRuntime bridgeRuntime) {
38          this.bridgeRuntime = bridgeRuntime;
39          config = bridgeRuntime.config();
40      }
41  
42      /**
43       * Retrieve the channel pipeline factory.
44       *
45       * @see org.jboss.netty.channel.ChannelPipelineFactory#getPipeline()
46       */
47      public ChannelPipeline getPipeline() {
48          final ChannelPipeline pipeline = pipeline();
49  
50          if (config.sslContext() != null) {
51              SSLEngine e = config.sslContext().createSSLEngine();
52              config.sslContextListener().onPostCreate(e);
53              pipeline.addLast("ssl", new SslHandler(e));
54          }
55  
56          if (config.nettySslContext() != null) {
57              pipeline.addLast("ssl", config.nettySslContext().newHandler());
58          }
59  
60          pipeline.addLast("decoder", new HttpRequestDecoder());
61  
62          if (config.aggregateRequestBodyInMemory()) {
63              pipeline.addLast("aggregator", new HttpChunkAggregator(config.maxChunkContentLength()));
64          }
65  
66          pipeline.addLast("encoder", new HttpResponseEncoder());
67  
68          if (config.supportChunking()) {
69              pipeline.addLast("chunkedWriter", new ChunkedWriteHandler());
70          }
71  
72          for (ChannelUpstreamHandler h: config.channelUpstreamHandlers()) {
73              pipeline.addLast(h.getClass().getName(), h);
74          }
75  
76          pipeline.addLast(BridgeRuntime.class.getName(), bridgeRuntime);
77  
78          return pipeline;
79      }
80  
81  }
