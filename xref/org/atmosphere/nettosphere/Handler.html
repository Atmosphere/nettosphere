16  package org.atmosphere.nettosphere;
17  
18  import org.atmosphere.cpr.AtmosphereResource;
19  
20  /**
21   * A super simple Handler for managing web application. As simple as
22   * <blockquote>
23   *    Nettosphere server = new Nettosphere.Builder().config(
24                   new Config.Builder()
25                      .host("127.0.0.1")
26                      .port(8080)
27                      .resource(new Handler() {
28                          void handle(AtmosphereResource r) {
29                              r.getResponse().write("Hello World".write("from Nettosphere").flush();
30                          }
31                      })
32                      .build())
33                   .build();
34      server.start();
35   * </blockquote>
36   */
37  public interface Handler {
38  
39      /**
40       * Handle an {@link AtmosphereResource}, from which you can retrieve an {@link org.atmosphere.cpr.AtmosphereRequest}
41       * and {@link }AtmosphereResponse}
42       * @param r an {@link AtmosphereResource}
43       */
44      void handle(AtmosphereResource r);
45  
46  }
