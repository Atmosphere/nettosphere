16  package org.atmosphere.nettosphere;
17  
18  import org.atmosphere.cpr.ApplicationConfig;
19  import org.atmosphere.cpr.AtmosphereConfig;
20  import org.atmosphere.websocket.WebSocket;
21  import org.jboss.netty.buffer.ChannelBuffer;
22  import org.jboss.netty.buffer.ChannelBufferFactory;
23  import org.jboss.netty.buffer.HeapChannelBufferFactory;
24  import org.jboss.netty.channel.Channel;
25  import org.jboss.netty.channel.ChannelFutureListener;
26  import org.jboss.netty.handler.codec.http.websocketx.BinaryWebSocketFrame;
27  import org.jboss.netty.handler.codec.http.websocketx.TextWebSocketFrame;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.io.IOException;
32  import java.util.concurrent.atomic.AtomicBoolean;
33  
34  public class NettyWebSocket extends WebSocket {
35  
36      private static final Logger logger = LoggerFactory.getLogger(NettyWebSocket.class);
37      private final Channel channel;
38      private final AtmosphereConfig config;
39      private final ChannelBufferFactory factory = new HeapChannelBufferFactory();
40      private final AtomicBoolean firstWrite = new AtomicBoolean(false);
41  
42      public NettyWebSocket(Channel channel, AtmosphereConfig config) {
43          this.channel = channel;
44          this.config = config;
45      }
46  
47      /**
48       * {@inheritDoc}
49       */
50      @Override
51      public WebSocket redirect(String location) throws IOException {
52          logger.error("redirect not supported");
53          return this;
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public WebSocket writeError(int errorCode, String message) throws IOException {
61         if (!firstWrite.get()) {
62              logger.debug("The WebSocket handshake succeeded but the dispatched URI failed {}:{}. " +
63                      "The WebSocket connection is still open and client can continue sending messages.", message, errorCode);
64          } else {
65              logger.debug("{} {}", errorCode, message);
66          }
67          return this;
68      }
69  
70      /**
71       * {@inheritDoc}
72       */
73      @Override
74      public WebSocket write(String data) throws IOException {
75          firstWrite.set(true);
76          if (channel.isOpen()) {
77              channel.write(new TextWebSocketFrame(data));
78          }
79          return this;
80      }
81  
82      /**
83       * {@inheritDoc}
84       */
85      @Override
86      public WebSocket write(byte[] data) throws IOException {
87          firstWrite.set(true);
88  
89          if (channel.isOpen()) {
90              String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_BINARY_WRITE);
91              if (s != null && Boolean.parseBoolean(s)) {
92                  ChannelBuffer c = factory.getBuffer(data.length);
93                  c.writeBytes(data);
94                  channel.write(new BinaryWebSocketFrame(c));
95              } else {
96                  channel.write(new TextWebSocketFrame(new String(data, 0, data.length, "UTF-8")));
97              }
98          }
99          return this;
100     }
101 
102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public WebSocket write(byte[] data, int offset, int length) throws IOException {
107         firstWrite.set(true);
108 
109         if (channel.isOpen()) {
110             String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_BINARY_WRITE);
111             if (s != null && Boolean.parseBoolean(s)) {
112                 ChannelBuffer c = factory.getBuffer(length - offset);
113                 c.writeBytes(data);
114                 channel.write(new BinaryWebSocketFrame(c));
115             } else {
116                 channel.write(new TextWebSocketFrame(new String(data, offset, length, "UTF-8")));
117             }
118         }
119         return this;
120     }
121 
122     /**
123      * {@inheritDoc}
124      */
125     @Override
126     public void close() throws IOException {
127         channel.close().addListener(ChannelFutureListener.CLOSE);
128     }
129 
130     /**
131      * {@inheritDoc}
132      */
133     @Override
134     public WebSocket flush() throws IOException {
135         return this;
136     }
137 }
