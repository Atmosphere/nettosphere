134     @Override
135     public void messageReceived(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
136         Object msg = messageEvent.getMessage();
137         if (msg instanceof HttpRequest) {
138             // Netty fail to decode headers separated by a ','
139             List<String> c = HttpRequest.class.cast(msg).getHeaders("Connection");
140             String u = HttpRequest.class.cast(msg).getHeader("Upgrade");
141             boolean webSocket = false;
142             if (u != null && u.equalsIgnoreCase("websocket")) {
143                 webSocket = true;
144             }
145 
146             for (String connection : c) {
147                 if (connection != null && connection.toLowerCase().equalsIgnoreCase("upgrade")) {
148                     webSocket = true;
149                 }
150             }
151             if (webSocket) {
152                 handleWebSocketHandshake(ctx, messageEvent);
153             } else {
154                 handleHttp(ctx, messageEvent);
155             }
156         } else if (msg instanceof WebSocketFrame) {
157             handleWebSocketFrame(ctx, messageEvent);
158         }
159     }

161     private void handleWebSocketHandshake(ChannelHandlerContext ctx, MessageEvent messageEvent) throws IOException, URISyntaxException {
162         final HttpRequest request = (HttpRequest) messageEvent.getMessage();
163 
164         // Allow only GET methods.
165         if (request.getMethod() != GET) {
166             sendHttpResponse(ctx, request, new DefaultHttpResponse(HTTP_1_1, FORBIDDEN));
167             return;
168         }
169 
170         WebSocketServerHandshakerFactory wsFactory = new WebSocketServerHandshakerFactory(
171                 getWebSocketLocation(request), null, false);
172         handshaker = wsFactory.newHandshaker(request);
173 
174         if (this.handshaker == null) {
175             wsFactory.sendUnsupportedWebSocketVersionResponse(ctx.getChannel());
176         } else {
177             this.handshaker.handshake(ctx.getChannel(), request);
178         }
179 
180         WebSocketProcessor processor = new WebSocketProcessor(framework,
181                 new NettyWebSocket(ctx.getChannel(), framework.getAtmosphereConfig()),
182                 framework.getWebSocketProtocol());
183         ctx.setAttachment(processor);
184         AtmosphereRequest r = createAtmosphereRequest(ctx, request);
185 
186         processor.dispatch(r);
187     }

189     private void handleWebSocketFrame(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
190         WebSocketFrame frame = (WebSocketFrame) messageEvent.getMessage();
191 
192         // Check for closing frame
193         if (frame instanceof CloseWebSocketFrame) {
194             this.handshaker.close(ctx.getChannel(), (CloseWebSocketFrame) frame);
195             return;
196         } else if (frame instanceof PingWebSocketFrame) {
197             ctx.getChannel().write(new PongWebSocketFrame(frame.getBinaryData()));
198             return;
199         } else if (!(frame instanceof TextWebSocketFrame)) {
200             throw new UnsupportedOperationException(String.format("%s frame types not supported", frame.getClass()
201                     .getName()));
202         }
203 
204         WebSocketProcessor processor = (WebSocketProcessor) ctx.getAttachment();
205         processor.invokeWebSocketProtocol(((TextWebSocketFrame) frame).getText());
206     }

208     private AtmosphereRequest createAtmosphereRequest(final ChannelHandlerContext ctx, HttpRequest request) throws URISyntaxException, UnsupportedEncodingException, MalformedURLException {
209         final String base = getBaseUri(request);
210         final URI requestUri = new URI(base.substring(0, base.length() - 1) + request.getUri());
211         String ct = request.getHeaders("Content-Type").size() > 0 ? request.getHeaders("Content-Type").get(0) : "text/plain";
212         String method = request.getMethod().getName();
213 
214         String queryString = requestUri.getQuery();
215         Map<String, String[]> qs = new HashMap<String, String[]>();
216         if (queryString != null) {
217             parseQueryString(qs, queryString);
218         }
219 
220         if (ct.equalsIgnoreCase("application/x-www-form-urlencoded")) {
221             parseQueryString(qs, new String(request.getContent().array(), "UTF-8"));
222         }
223 
224         String u = requestUri.toURL().toString();
225         int last = u.indexOf("?") == -1 ? u.length() : u.indexOf("?");
226         String url = u.substring(0, last);
227         int l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3;
228         final Map<String, Object> attributes = new HashMap<String, Object>();
229 
230         AtmosphereRequest.Builder requestBuilder = new AtmosphereRequest.Builder();
231         AtmosphereRequest r = requestBuilder.requestURI(url.substring(l))
232                 .requestURL(url)
233                 .pathInfo(url.substring(l))
234                 .headers(getHeaders(request))
235                 .method(method)
236                 .contentType(ct)
237                 .attributes(attributes)
238                 .queryStrings(qs)
239                 .remotePort(((InetSocketAddress) ctx.getChannel().getRemoteAddress()).getPort())
240                 .remoteAddr(((InetSocketAddress) ctx.getChannel().getRemoteAddress()).getAddress().getHostAddress())
241                 .remoteHost(((InetSocketAddress) ctx.getChannel().getRemoteAddress()).getHostName())
242                 .localPort(((InetSocketAddress) ctx.getChannel().getLocalAddress()).getPort())
243                 .localAddr(((InetSocketAddress) ctx.getChannel().getLocalAddress()).getAddress().getHostAddress())
244                 .localName(((InetSocketAddress) ctx.getChannel().getLocalAddress()).getHostName())
245                 .inputStream(new ChannelBufferInputStream(request.getContent()))
246                 .build();
247 
248         return r;
249     }

251     private void handleHttp(final ChannelHandlerContext ctx, final MessageEvent messageEvent) throws URISyntaxException, IOException {
252         final ChannelAsyncIOWriter w = new ChannelAsyncIOWriter(ctx.getChannel());
253         boolean resumeOnBroadcast = false;
254         boolean keptOpen = false;
255         final HttpRequest request = (HttpRequest) messageEvent.getMessage();
256         String method = request.getMethod().getName();
257 
258         // First let's try to see if it's a static resources
259         try {
260             request.addHeader(STATIC_MAPPING, "true");
261             super.messageReceived(ctx, messageEvent);
262 
263             if (request.getHeader(SERVICED) != null) {
264                 return;
265             }
266         } catch (Exception e) {
267             logger.debug("", e);
268         } finally {
269             request.addHeader(STATIC_MAPPING, "false");
270         }
271 
272         boolean skipClose = false;
273         try {
274             AtmosphereRequest r = createAtmosphereRequest(ctx, request);
275 
276             AtmosphereResponse response = new AtmosphereResponse.Builder()
277                     .writeHeader(true)
278                     .asyncIOWriter(w)
279                     .header("Connection", "Keep-Alive")
280                     .header("Transfer-Encoding", "chunked")
281                     .header("Server", "Nettosphere-" +  Version.getRawVersion())
282                     .request(r).build();
283 
284             r.setAttribute(NettyCometSupport.CHANNEL, w);
285 
286             framework.doCometSupport(r, response);
287 
288             final AsynchronousProcessor.AsynchronousProcessorHook hook = (AsynchronousProcessor.AsynchronousProcessorHook) r.getAttribute(FrameworkConfig.ASYNCHRONOUS_HOOK);
289             ctx.setAttachment(hook);
290 
291             String transport = (String) r.getAttribute(FrameworkConfig.TRANSPORT_IN_USE);
292             if (transport != null && transport.equalsIgnoreCase(HeaderConfig.STREAMING_TRANSPORT)) {
293                 keptOpen = true;
294             } else if (transport != null && transport.equalsIgnoreCase(HeaderConfig.LONG_POLLING_TRANSPORT)) {
295                 resumeOnBroadcast = true;
296             }
297 
298             final AtmosphereFramework.Action action = (AtmosphereFramework.Action) r.getAttribute(NettyCometSupport.SUSPEND);
299             if (action != null && action.type == AtmosphereFramework.Action.TYPE.SUSPEND && action.timeout != -1) {
300                 final AtomicReference<Future<?>> f = new AtomicReference();
301                 f.set(suspendTimer.scheduleAtFixedRate(new Runnable() {
302                     @Override
303                     public void run() {
304                         if (!w.isClosed() && (System.currentTimeMillis() - w.lastTick()) > action.timeout) {
305                             hook.timedOut();
306                             f.get().cancel(true);
307                         }
308                     }
309                 }, action.timeout, action.timeout, TimeUnit.MILLISECONDS));
310             } else if (action != null && action.type == AtmosphereFramework.Action.TYPE.RESUME) {
311                 resumeOnBroadcast = false;
312             }
313             w.resumeOnBroadcast(resumeOnBroadcast);
314         } catch (AtmosphereMappingException ex) {
315             if (method.equalsIgnoreCase("GET")) {
316                 logger.trace("Unable to map the request {}, trying static file", messageEvent.getMessage());
317                 try {
318                     skipClose = true;
319                     super.messageReceived(ctx, messageEvent);
320                 } catch (Exception e) {
321                     logger.error("Unable to process request", e);
322                     throw new IOException(e);
323                 }
324             }
325         } catch (Throwable e) {
326             logger.error("Unable to process request", e);
327             throw new IOException(e);
328         } finally {
329             if (w != null && !resumeOnBroadcast && !keptOpen) {
330                 if (!w.byteWritten()) {
331                     w.writeError(200, "OK");
332                 }
333                 if (!skipClose) {
334                     w.close();
335                 }
336             }
337         }
338     }

353     @Override
354     public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent e) throws Exception {
355         super.channelClosed(ctx, e);
356         Object o = ctx.getAttachment();
357         if (o == null) return;
358 
359         if (AsynchronousProcessor.AsynchronousProcessorHook.class.isAssignableFrom(o.getClass())) {
360             AsynchronousProcessor.AsynchronousProcessorHook hook = AsynchronousProcessor.AsynchronousProcessorHook.class.cast(o);
361             if (hook != null) {
362                 hook.closed();
363             }
364         } else if (WebSocketProcessor.class.isAssignableFrom(o.getClass())) {
365             AsynchronousProcessor.AsynchronousProcessorHook hook = AsynchronousProcessor.AsynchronousProcessorHook.class.cast(
366                     WebSocketProcessor.class.cast(o).webSocket().resource().getRequest().getAttribute(FrameworkConfig.ASYNCHRONOUS_HOOK));
367             if (hook != null) {
368                 hook.closed();
369             }
370             WebSocketProcessor.class.cast(o).close(1000);
371         }
372     }

381     private Map<String, String> getHeaders(final HttpRequest request) {
382         final Map<String, String> headers = new HashMap<String, String>();

397     private void parseQueryString(Map<String, String[]> qs, String queryString) {
398         if (queryString != null) {
399             String[] s = queryString.split("&");
400             for (String a : s) {
401                 String[] q = a.split("=");
402                 String[] z = new String[]{q.length > 1 ? q[1] : ""};
403                 qs.put(q[0], z);
404             }
405         }
406     }

408     private void sendHttpResponse(ChannelHandlerContext ctx, HttpRequest req, HttpResponse res) {
409         // Generate an error page if response status code is not OK (200).
410         if (res.getStatus().getCode() != 200) {
411             res.setContent(ChannelBuffers.copiedBuffer(res.getStatus().toString(), CharsetUtil.UTF_8));
412             setContentLength(res, res.getContent().readableBytes());
413         }
414 
415         // Send the response and close the connection if necessary.
416         ChannelFuture f = ctx.getChannel().write(res);
417         if (!isKeepAlive(req) || res.getStatus().getCode() != 200) {
418             f.addListener(ChannelFutureListener.CLOSE);
419         }
420     }
