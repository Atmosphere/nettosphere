179     /**
180      * @param extension the extension
181      * @return the content type associated with <code>extension</code>.  If
182      *         no association is found, this method will return <code>text/plain</code>
183      */
184     public static String get(String extension) {
185         return contentTypes.getProperty(extension, "text/plain");
186     }
187 
188 
189     /**
190      * @param extension the extension
191      * @param defaultCt the content type to return if there is no known
192      *                  association for the specified extension
193      * @return the content type associated with <code>extension</code>
194      *         or if no associate is found, returns <code>defaultCt</code>
195      */
196     public static String get(String extension, String defaultCt) {
197         return contentTypes.getProperty(extension, defaultCt);
198     }
199 
200 
201     /**
202      * @param extension the extension
203      * @return <code>true</code> if the specified extension has been registered
204      *         otherwise, returns <code>false</code>
205      */
206     public static boolean contains(String extension) {
207         return (contentTypes.getProperty(extension) != null);
208     }
209 
210 
211     /**
212      * <p>
213      * Associates the specified extension and content type
214      * </p>
215      *
216      * @param extension   the extension
217      * @param contentType the content type associated with the extension
218      */
219     public static void add(String extension, String contentType) {
220         if (extension == null
221                 || extension.length() == 0
222                 || contentType == null
223                 || contentType.length() == 0) {
224             return;
225         }
226         contentTypes.put(extension, contentType);
227     }
