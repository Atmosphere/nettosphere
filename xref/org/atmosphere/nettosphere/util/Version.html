54  package org.atmosphere.nettosphere.util;
55  
56  import java.io.IOException;
57  import java.io.InputStream;
58  import java.util.Properties;
59  import java.util.regex.Matcher;
60  import java.util.regex.Pattern;
61  
62  /**
63   * Class contains information about Atmosphere framework
64   *
65   * @author Charlie Hunt
66   * @author Hubert Iwaniuk
67   */
68  public class Version {
69  
70      private static final Pattern versionPattern = Pattern.compile("((\\d+)\\.(\\d+)\\.(\\d+)){1}(.+)?");
71      private static final String dotedVersion;
72      private static final int major;
73      private static final int minor;
74      private static final int micro;
75      private static final String version;
76  
77      public static void main(String[] args) {
78          System.out.println(Version.getDotedVersion());
79      }
80  
81      /** Reads version from properties and parses it. */
82      static {
83          Properties prop = new Properties();
84          InputStream s = null;
85          try {
86              s = Version.class.getResourceAsStream("version.properties");
87              prop.load(s);
88          } catch (IOException e) {
89              e.printStackTrace();
90          } finally {
91              if (s != null) {
92                  try {
93                      s.close();
94                  } catch (IOException ex) {
95                  }
96              }
97          }
98          version = prop.getProperty("nettosphere.version");
99          Matcher matcher = versionPattern.matcher(version);
100         if (matcher.matches()) {
101             dotedVersion = matcher.group(1);
102             major = Integer.parseInt(matcher.group(2));
103             minor = Integer.parseInt(matcher.group(3));
104             micro = Integer.parseInt(matcher.group(4));
105         } else {
106             dotedVersion = "no.version";
107             major = -1;
108             minor = -1;
109             micro = -1;
110         }
111     }
112 
113     /**
114      * Return the raw version derived frok the project's pom.xml
115      *
116      * @return
117      */
118     public static String getRawVersion() {
119         return version;
120     }
121 
122     /**
123      * Return the dotted version of the curent release.
124      *
125      * @return like "2.0.1"
126      */
127     public static String getDotedVersion() {
128         return dotedVersion;
129     }
130 
131 
132     /**
133      * Get Atmosphere framework major version
134      *
135      * @return Atmosphere framework major version
136      */
137     public static int getMajorVersion() {
138         return major;
139     }
140 
141     /**
142      * Get Atmosphere framework minor version
143      *
144      * @return Atmosphere framework minor version
145      */
146     public static int getMinorVersion() {
147         return minor;
148     }
149 
150     /**
151      * Return the micro version
152      */
153     public static int getMicroVersion() {
154         return micro;
155     }
156 
157     /**
158      * Checks if current Atmosphere framework version equals to one passed
159      *
160      * @param major Atmosphere framework major version
161      * @param minor Atmosphere framework minor version
162      * @return true, if versions are equal; false otherwise
163      */
164     public static boolean equalVersion(int major, int minor) {
165         return minor == Version.minor && major == Version.major;
166     }
