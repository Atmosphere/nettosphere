16  package org.atmosphere.nettosphere.util;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.util.Properties;
21  import java.util.regex.Matcher;
22  import java.util.regex.Pattern;
23  
24  /**
25   * Class contains information about Atmosphere framework
26   *
27   * @author Charlie Hunt
28   * @author Hubert Iwaniuk
29   */
30  public class Version {
31  
32      private static final Pattern versionPattern = Pattern.compile("((\\d+)\\.(\\d+)\\.(\\d+)){1}(.+)?");
33      private static final String dotedVersion;
34      private static final int major;
35      private static final int minor;
36      private static final int micro;
37      private static final String version;
38  
39      public static void main(String[] args) {
40          System.out.println(Version.getDotedVersion());
41      }
42  
43      /** Reads version from properties and parses it. */
44      static {
45          Properties prop = new Properties();
46          InputStream s = null;
47          try {
48              s = Version.class.getResourceAsStream("version.properties");
49              prop.load(s);
50          } catch (IOException e) {
51              e.printStackTrace();
52          } finally {
53              if (s != null) {
54                  try {
55                      s.close();
56                  } catch (IOException ex) {
57                  }
58              }
59          }
60          version = prop.getProperty("nettosphere.version");
61          Matcher matcher = versionPattern.matcher(version);
62          if (matcher.matches()) {
63              dotedVersion = matcher.group(1);
64              major = Integer.parseInt(matcher.group(2));
65              minor = Integer.parseInt(matcher.group(3));
66              micro = Integer.parseInt(matcher.group(4));
67          } else {
68              dotedVersion = "no.version";
69              major = -1;
70              minor = -1;
71              micro = -1;
72          }
73      }
74  
75      /**
76       * Return the raw version derived frok the project's pom.xml
77       *
78       * @return
79       */
80      public static String getRawVersion() {
81          return version;
82      }
83  
84      /**
85       * Return the dotted version of the curent release.
86       *
87       * @return like "2.0.1"
88       */
89      public static String getDotedVersion() {
90          return dotedVersion;
91      }
92  
93  
94      /**
95       * Get Atmosphere framework major version
96       *
97       * @return Atmosphere framework major version
98       */
99      public static int getMajorVersion() {
100         return major;
101     }
102 
103     /**
104      * Get Atmosphere framework minor version
105      *
106      * @return Atmosphere framework minor version
107      */
108     public static int getMinorVersion() {
109         return minor;
110     }
111 
112     /**
113      * Return the micro version
114      */
115     public static int getMicroVersion() {
116         return micro;
117     }
118 
119     /**
120      * Checks if current Atmosphere framework version equals to one passed
121      *
122      * @param major Atmosphere framework major version
123      * @param minor Atmosphere framework minor version
124      * @return true, if versions are equal; false otherwise
125      */
126     public static boolean equalVersion(int major, int minor) {
127         return minor == Version.minor && major == Version.major;
128     }
