16  package org.atmosphere.nettosphere.util;

18  import javax.net.ssl.SSLEngine;

20  /**
21   * A callback used to configure {@link javax.net.ssl.SSLEngine} before they get injected in Netty.
22   */
23  public interface SSLContextListener {
24  
25      SSLContextListener DEFAULT = new SSLContextListener(){
26  
27          @Override
28          public void onPostCreate(SSLEngine e) {
29              e.setEnabledCipherSuites(new String[]{"SSL_DH_anon_WITH_RC4_128_MD5"});
30              e.setUseClientMode(false);
31          }
32      };
33  
34      /**
35       * Invoked just after the {@link SSLEngine} has been created, but not yet injected in Netty.
36       * @param e SSLEngine;
37       */
38      public void onPostCreate(SSLEngine e);

40  }
