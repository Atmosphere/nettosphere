16  package org.atmosphere.nettosphere.util;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.atmosphere.util.ExecutorsFactory;
20  import org.jboss.netty.buffer.ChannelBuffer;
21  import org.jboss.netty.buffer.ChannelBuffers;
22  
23  import java.util.concurrent.ConcurrentLinkedQueue;
24  import java.util.concurrent.TimeUnit;
25  
26  public class ChannelBufferPool {
27      private ConcurrentLinkedQueue<ChannelBuffer> pool;
28  
29      public ChannelBufferPool(final int minIdle) {
30          initialize(minIdle);
31      }
32  
33      public ChannelBufferPool(final int minIdle, final int maxIdle, final long validationInterval, AtmosphereConfig config) {
34          initialize(minIdle);
35  
36          ExecutorsFactory.getScheduler(config).scheduleWithFixedDelay(new Runnable() {
37              @Override
38              public void run() {
39                  int size = pool.size();
40                  if (size < minIdle) {
41                      int sizeToBeAdded = minIdle - size;
42                      for (int i = 0; i < sizeToBeAdded; i++) {
43                          pool.add(createObject());
44                      }
45                  } else if (size > maxIdle) {
46                      int sizeToBeRemoved = size - maxIdle;
47                      for (int i = 0; i < sizeToBeRemoved; i++) {
48                          pool.poll();
49                      }
50                  }
51              }
52          }, validationInterval, validationInterval, TimeUnit.SECONDS);
53  
54          config.shutdownHook(new AtmosphereConfig.ShutdownHook() {
55              @Override
56              public void shutdown() {
57                  pool.clear();
58              }
59          });
60      }
61  
62      public ChannelBuffer poll() {
63          ChannelBuffer channelBuffer;
64          if ((channelBuffer = pool.poll()) == null) {
65              channelBuffer = createObject();
66          }
67  
68          return channelBuffer;
69      }
70  
71      public void offer(ChannelBuffer channelBuffer) {
72          if (channelBuffer == null) {
73              return;
74          }
75  
76          channelBuffer.clear();
77          this.pool.offer(channelBuffer);
78      }
79  
80      protected ChannelBuffer createObject() {
81          return ChannelBuffers.dynamicBuffer();
82      }
83  
84      private void initialize(final int minIdle) {
85          pool = new ConcurrentLinkedQueue<ChannelBuffer>();
86  
87          for (int i = 0; i < minIdle; i++) {
88              pool.add(createObject());
89          }
90      }
91  }
