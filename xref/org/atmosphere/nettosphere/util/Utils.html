34      public final static URLClassLoader createURLClassLoader(String dirPath) throws IOException {
35  
36          String path;
37          File file;
38          URL appRoot;
39          URL classesURL;
40  
41          if (!dirPath.endsWith("/") &&
42                  !dirPath.endsWith(".war") &&
43                  !dirPath.endsWith(".jar")) {
44              dirPath += File.separator;
45          }
46  
47          //Must be a better way because that sucks!
48          String separator = (System.getProperty("os.name").toLowerCase().startsWith("win") ? "/" : "//");
49  
50          if (dirPath != null &&
51                  (dirPath.endsWith(".war") || dirPath.endsWith(".jar"))) {
52              file = new File(dirPath);
53              appRoot = new URL("jar:file:" + separator +
54                      file.getCanonicalPath().replace('\\', '/') + "!/");
55              classesURL = new URL("jar:file:" + separator +
56                      file.getCanonicalPath().replace('\\', '/') + "!/WEB-INF/classes/");
57  
58              path = expand(appRoot);
59  
60              //DEBUG
61              //classesURL = new URL("file:/" + path + "WEB-INF/classes/");
62              //appRoot = new URL("file:/" + path);
63  
64          } else {
65              path = dirPath;
66              classesURL = new URL("file://" + path + "WEB-INF/classes/");
67              appRoot = new URL("file://" + path);
68          }
69  
70          String absolutePath = new File(path).getAbsolutePath();
71          URL[] urls;
72          File libFiles = new File(absolutePath + File.separator + "WEB-INF" + File.separator + "lib");
73          int arraySize = 4;
74  
75          if (libFiles.exists() && libFiles.isDirectory()) {
76              urls = new URL[libFiles.listFiles().length + arraySize];
77              for (int i = 0; i < libFiles.listFiles().length; i++) {
78                  urls[i] = new URL("jar:file:" + separator + libFiles.listFiles()[i].toString().replace('\\', '/') + "!/");
79              }
80          } else {
81              urls = new URL[arraySize];
82          }
83  
84          urls[urls.length - 1] = classesURL;
85          urls[urls.length - 2] = appRoot;
86          urls[urls.length - 3] = new URL("file://" + path + "/WEB-INF/classes/");
87          urls[urls.length - 4] = new URL("file://" + path);
88  
89          return new URLClassLoader(urls, Thread.currentThread().getContextClassLoader());
90  
91      }

109     /**
110      * Expand the jar file found at the specified URL into an unpacked
111      * directory structure, and return the absolute pathname to the expanded
112      * directory.
113      *
114      * @param jar        URL of the web application archive to be expanded
115      *                   (must start with "jar:")
116      * @param workFolder the folder where the file will be expanded
117      * @return Absolute path as in {@link java.io.File#getAbsolutePath()} of location where to find expanded jar.
118      * @throws IllegalArgumentException if this is not a "jar:" URL
119      * @throws IOException              if an input/output error was encountered
120      *                                  during expansion
121      */
122     public static String expand(URL jar, String workFolder) throws IOException {
123 
124         // Calculate the directory name of the expanded directory
125         String pathname = jar.toString().replace('\\', '/');
126         if (pathname.endsWith("!/")) {
127             pathname = pathname.substring(0, pathname.length() - 2);
128         }
129         int period = pathname.lastIndexOf('.');
130         if (period >= pathname.length() - 4)
131             pathname = pathname.substring(0, period);
132         int slash = pathname.lastIndexOf('/');
133         if (slash >= 0) {
134             pathname = pathname.substring(slash + 1);
135         }
136         return expand(jar, pathname, workFolder);
137 
138     }

141     /**
142      * Expand the jar file found at the specified URL into an unpacked
143      * directory structure, and return the absolute pathname to the expanded
144      * directory.
145      *
146      * @param jar      URL of the web application archive to be expanded
147      *                 (must start with "jar:")
148      * @param pathname Context path name for web application
149      * @return Absolute path as in {@link java.io.File#getAbsolutePath()} of location where to find expanded jar.
150      * @throws IllegalArgumentException if this is not a "jar:" URL
151      * @throws IOException              if an input/output error was encountered
152      *                                  during expansion
153      */
154     public static String expand(URL jar, String pathname, String dirname)
155             throws IOException {
156 
157         // Make sure that there is no such directory already existing
158         File appBase = new File(dirname);
159         File docBase = new File(appBase, pathname);
160         // Create the new document base directory
161         if (!docBase.mkdir()) {
162             throw new IllegalStateException(String.format("Unable to create directory: %s", docBase.getAbsolutePath()));
163         }
164 
165         // Expand the jar into the new document base directory
166         JarURLConnection juc = (JarURLConnection) jar.openConnection();
167         juc.setUseCaches(false);
168         JarFile jarFile = null;
169         InputStream input = null;
170         try {
171             jarFile = juc.getJarFile();
172             Enumeration<JarEntry> jarEntries = jarFile.entries();
173             while (jarEntries.hasMoreElements()) {
174                 JarEntry jarEntry = jarEntries.nextElement();
175                 String name = jarEntry.getName();
176                 int last = name.lastIndexOf('/');
177                 if (last >= 0) {
178                     File parent = new File(docBase,
179                             name.substring(0, last));
180                     if (!parent.mkdirs()) {
181                         throw new IllegalStateException(String.format("Unable to create directory: %s", parent.getAbsolutePath()));
182                     }
183                 }
184                 if (name.endsWith("/")) {
185                     continue;
186                 }
187                 input = jarFile.getInputStream(jarEntry);
188                 expand(input, docBase, name);
189                 input.close();
190                 input = null;
191             }
192         } catch (IOException e) {
193             // If something went wrong, delete expanded dir to keep things
194             // clean
195             deleteDir(docBase);
196             throw e;
197         } finally {
198             if (input != null) {
199                 try {
200                     input.close();
201                 } catch (Throwable ignored) {
202                 }
203                 input = null;
204             }
205             if (jarFile != null) {
206                 try {
207                     jarFile.close();
208                 } catch (Throwable ignored) {
209                 }
210                 jarFile = null;
211             }
212         }
213 
214         // Return the absolute path to our new document base directory
215         return (docBase.getAbsolutePath());
216 
217     }

219     /**
220      * Expand the specified input stream into the specified directory, creating
221      * a file named from the specified relative path.
222      *
223      * @param input   InputStream to be copied
224      * @param docBase Document base directory into which we are expanding
225      * @param name    Relative pathname of the file to be created
226      * @throws IOException if an input/output error occurs
227      */
228     protected static void expand(InputStream input, File docBase, String name)
229             throws IOException {
230 
231         File file = new File(docBase, name);
232         BufferedOutputStream output = null;
233         try {
234             output =
235                     new BufferedOutputStream(new FileOutputStream(file));
236             byte buffer[] = new byte[2048];
237             while (true) {
238                 int n = input.read(buffer);
239                 if (n <= 0)
240                     break;
241                 output.write(buffer, 0, n);
242             }
243         } finally {
244             if (output != null) {
245                 try {
246                     output.close();
247                 } catch (IOException ignored) {
248                     // Ignore
249                 }
250             }
251         }
252 
253     }

255     /**
256      * Delete the specified directory, including all of its contents and
257      * subdirectories recursively.
258      *
259      * @param dir File object representing the directory to be deleted
260      */
261     public static boolean deleteDir(File dir) {
262 
263         String files[] = dir.list();
264         if (files == null) {
265             files = new String[0];
266         }
267         for (int i = 0; i < files.length; i++) {
268             File file = new File(dir, files[i]);
269             if (file.isDirectory()) {
270                 deleteDir(file);
271             } else {
272                 if (!file.delete()) {
273                     throw new IllegalStateException(String.format("Unable to delete file: %s", file.getAbsolutePath()));
274                 }
275             }
276         }
277         return dir.delete();
278 
279     }

281     /**
282      * Copy the specified file or directory to the destination.
283      *
284      * @param src  File object representing the source
285      * @param dest File object representing the destination
286      */
287     public static boolean copy(File src, File dest) {
288 
289         boolean result = true;
290 
291         String files[] = null;
292         if (src.isDirectory()) {
293             files = src.list();
294             result = dest.mkdir();
295         } else {
296             files = new String[1];
297             files[0] = "";
298         }
299         if (files == null) {
300             files = new String[0];
301         }
302         for (int i = 0; (i < files.length) && result; i++) {
303             File fileSrc = new File(src, files[i]);
304             File fileDest = new File(dest, files[i]);
305             if (fileSrc.isDirectory()) {
306                 result = copy(fileSrc, fileDest);
307             } else {
308                 FileChannel ic = null;
309                 FileChannel oc = null;
310                 try {
311                     ic = (new FileInputStream(fileSrc)).getChannel();
312                     oc = (new FileOutputStream(fileDest)).getChannel();
313                     ic.transferTo(0, ic.size(), oc);
314                 } catch (IOException e) {
315 
316                     result = false;
317                 } finally {
318                     if (ic != null) {
319                         try {
320                             ic.close();
321                         } catch (IOException ignored) {
322                         }
323                     }
324                     if (oc != null) {
325                         try {
326                             oc.close();
327                         } catch (IOException ignored) {
328                         }
329                     }
330                 }
331             }
332         }
333         return result;
