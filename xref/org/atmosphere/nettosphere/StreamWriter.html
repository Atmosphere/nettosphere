16  package org.atmosphere.nettosphere;
17  
18  import org.atmosphere.cpr.AsyncIOWriter;
19  import org.atmosphere.cpr.AtmosphereResponse;
20  import org.jboss.netty.buffer.ChannelBuffer;
21  import org.jboss.netty.buffer.ChannelBuffers;
22  import org.jboss.netty.channel.Channel;
23  import org.jboss.netty.channel.ChannelFuture;
24  import org.jboss.netty.channel.ChannelFutureListener;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import java.io.IOException;
29  
30  /**
31   * A streaming {@link ChannelWriter} that write unchuncked bytes.
32   */
33  public class StreamWriter extends ChannelWriter {
34      private static final Logger logger = LoggerFactory.getLogger(StreamWriter.class);
35      private ChannelBuffer chainedBodyBuffer;
36  
37      public StreamWriter(Channel channel, boolean writeHeader, boolean keepAlive) {
38          super(channel, writeHeader, keepAlive);
39          setUpBuffers();
40      }
41  
42      @Override
43      public AsyncIOWriter asyncWrite(AtmosphereResponse response, byte[] data, int offset, int length) throws IOException {
44          chainedBodyBuffer = ChannelBuffers.wrappedBuffer(chainedBodyBuffer, ChannelBuffers.wrappedBuffer(data, offset, length));
45          lastWrite = System.currentTimeMillis();
46          return this;
47      }
48  
49      private void setUpBuffers() {
50          if (chainedBodyBuffer == null) {
51              chainedBodyBuffer = ChannelBuffers.EMPTY_BUFFER;
52          }
53      }
54  
55      @Override
56      public void close(AtmosphereResponse response) throws IOException {
57          // Once we initiate the flush, we discard anything coming after for whatever reason.
58          if (!doneProcessing.getAndSet(true) && channel.isOpen()) {
59              logger.trace("About to flush to {} for {}", channel, response.uuid());
60  
61              ChannelBuffer statusAndHeadersBuffer = writeHeader ?
62                      ChannelBuffers.wrappedBuffer(constructStatusAndHeaders(response, chainedBodyBuffer.readableBytes()).getBytes("UTF-8")) : ChannelBuffers.EMPTY_BUFFER;
63              ChannelBuffer drain = ChannelBuffers.wrappedBuffer(statusAndHeadersBuffer, chainedBodyBuffer);
64              channel.write(drain).addListener(new ChannelFutureListener() {
65                  @Override
66                  public void operationComplete(ChannelFuture channelFuture) throws Exception {
67                      chainedBodyBuffer = null;
68                      if (!keepAlive) {
69                          channel.close().awaitUninterruptibly();
70                      }
71                  }
72              });
73          } else {
74              throw new IOException(channel + ": content already processed for " + response.uuid());
75          }
76      }
77  
78  }
