23  import io.netty.channel.Channel;
24  import io.netty.channel.ChannelFutureListener;
25  import io.netty.handler.codec.http.DefaultHttpResponse;
26  import io.netty.handler.codec.http.HttpResponseStatus;
27  import io.netty.handler.codec.http.HttpVersion;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;

31  import java.io.IOException;
32  import java.util.Map;
33  import java.util.concurrent.atomic.AtomicBoolean;

35  public abstract class ChannelWriter extends AtmosphereInterceptorWriter {
36      protected final static Logger logger = LoggerFactory.getLogger(StreamWriter.class);
37      protected final static String END = Integer.toHexString(0);
38      protected static final String CRLF = "\r\n";
39      protected final static byte[] CHUNK_DELIMITER = CRLF.getBytes();
40      protected final static byte[] ENDCHUNK = (END + CRLF + CRLF).getBytes();
41  
42      protected final Channel channel;
43      protected final AtomicBoolean doneProcessing = new AtomicBoolean(false);
44      protected final ByteArrayAsyncWriter transformCacheBuffer = new ByteArrayAsyncWriter();
45      protected final boolean writeHeader;
46  
47      protected long lastWrite = 0;
48      protected boolean keepAlive;
49  
50      public ChannelWriter(Channel channel, boolean writeHeader, boolean keepAlive) {
51          this.channel = channel;
52          this.writeHeader = writeHeader;
53          this.keepAlive = keepAlive;
54      }
55  
56      public boolean isClosed() {
57          return doneProcessing.get();
58      }
59  
60      @Override
61      public AsyncIOWriter writeError(AtmosphereResponse response, int errorCode, String message) throws IOException {
62          if (!channel.isOpen()) {
63              return this;
64          }
65  
66          try {
67              DefaultHttpResponse r = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
68                      HttpResponseStatus.valueOf(errorCode));
69              channel.write(r).addListener(ChannelFutureListener.CLOSE);
70          } catch (Throwable ex) {
71              logger.debug("", ex);
72          }
73          return this;
74      }
75  
76      @Override
77      public AsyncIOWriter write(AtmosphereResponse response, String data) throws IOException {
78          byte[] b = data.getBytes("ISO-8859-1");
79          write(response, b);
80          return this;
81      }
82  
83      @Override
84      public AsyncIOWriter write(AtmosphereResponse response, byte[] data) throws IOException {
85          write(response, data, 0, data.length);
86          return this;
87      }
88  
89      protected byte[] transform(AtmosphereResponse response, byte[] b, int offset, int length) throws IOException {
90          AsyncIOWriter a = response.getAsyncIOWriter();
91          try {
92              response.asyncIOWriter(transformCacheBuffer);
93              invokeInterceptor(response, b, offset, length);
94              return transformCacheBuffer.stream().toByteArray();
95          } finally {
96              transformCacheBuffer.close(null);
97              response.asyncIOWriter(a);
98          }
99      }
100 
101     @Override
102     public AsyncIOWriter write(AtmosphereResponse response, byte[] data, int offset, int length) throws IOException {
103 
104         if (doneProcessing.get()) {
105             throw Utils.ioExceptionForChannel(channel, response.uuid());
106         }
107 
108         boolean transform = filters.size() > 0 && response.getStatus() < 400;
109         if (transform) {
110             data = transform(response, data, offset, length);
111             offset = 0;
112             length = data.length;
113         }
114 
115         logger.trace("About to write to {}", channel);
116         if (channel.isOpen()) {
117             asyncWrite(response, data, offset, length);
118         } else {
119             logger.trace("Trying to write on a closed channel {}", channel);
120             throw new IOException("Channel closed");
121         }
122         return this;
123     }
124 
125 
126     public long lastTick() {
127         return lastWrite == -1 ? System.currentTimeMillis() : lastWrite;
128     }
129 
130     protected String constructStatusAndHeaders(AtmosphereResponse response, int contentLength) {
131         StringBuffer b = new StringBuffer("HTTP/1.1")
132                 .append(" ")
133                 .append(response.getStatus())
134                 .append(" ")
135                 .append(response.getStatusMessage())
136                 .append(CRLF);
137 
138         Map<String, String> headers = response.headers();
139         String contentType = response.getContentType();
140 
141         b.append("Content-Type").append(":").append(headers.get("Content-Type") == null ? contentType : headers.get("Content-Type")).append(CRLF);
142         if (contentLength != -1) {
143             b.append("Content-Length").append(":").append(contentLength).append(CRLF);
144         }
145 
146         for (String s : headers.keySet()) {
147             if (!s.equalsIgnoreCase("Content-Type")) {
148                 b.append(s).append(":").append(headers.get(s)).append(CRLF);
149             }
150         }
151         b.append(CRLF);
152         return b.toString();
153     }
154 
155     abstract public AsyncIOWriter asyncWrite(AtmosphereResponse response, byte[] data, int offset, int length) throws IOException;
156 
157 }
