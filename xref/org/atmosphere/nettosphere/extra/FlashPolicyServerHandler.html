16  package org.atmosphere.nettosphere.extra;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.buffer.ChannelBuffers;
20  import org.jboss.netty.channel.ChannelFuture;
21  import org.jboss.netty.channel.ChannelFutureListener;
22  import org.jboss.netty.channel.ChannelHandlerContext;
23  import org.jboss.netty.channel.ExceptionEvent;
24  import org.jboss.netty.channel.MessageEvent;
25  import org.jboss.netty.channel.SimpleChannelUpstreamHandler;
26  import org.jboss.netty.handler.timeout.ReadTimeoutException;
27  import org.jboss.netty.util.CharsetUtil;
28  
29  /**
30   * @author <a href="http://www.waywardmonkeys.com/">Bruce Mitchener</a>
31   */
32  public class FlashPolicyServerHandler extends SimpleChannelUpstreamHandler {
33  
34      private static final String NEWLINE = "\r\n";
35  
36      @Override
37      public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
38  //        Object msg = e.getMessage();
39          ChannelFuture f = e.getChannel().write(this.getPolicyFileContents());
40          f.addListener(ChannelFutureListener.CLOSE);
41      }
42  
43      private ChannelBuffer getPolicyFileContents() throws Exception {
44          return ChannelBuffers.copiedBuffer(
45              "<?xml version=\"1.0\"?>" + NEWLINE +
46              "<!DOCTYPE cross-domain-policy SYSTEM \"/xml/dtds/cross-domain-policy.dtd\">" + NEWLINE +
47              "" + NEWLINE +
48              "<!-- Policy file for xmlsocket://socks.example.com -->" + NEWLINE +
49              "<cross-domain-policy> " + NEWLINE +
50              "" + NEWLINE +
51              "   <!-- This is a master socket policy file -->" + NEWLINE +
52              "   <!-- No other socket policies on the host will be permitted -->" + NEWLINE +
53              "   <site-control permitted-cross-domain-policies=\"master-only\"/>" + NEWLINE +
54              "" + NEWLINE +
55              "   <!-- Instead of setting to-ports=\"*\", administrator's can use ranges and commas -->" + NEWLINE +
56              "   <allow-access-from domain=\"*\" to-ports=\"8080\" />" + NEWLINE +
57              "" + NEWLINE +
58              "</cross-domain-policy>" + NEWLINE,
59              CharsetUtil.US_ASCII);
60      }
61  
62      @Override
63      public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
64              throws Exception {
65          if (e.getCause() instanceof ReadTimeoutException) {
66              System.out.println("Connection timed out.");
67              e.getChannel().close();
68          } else {
69              e.getCause().printStackTrace();
70              e.getChannel().close();
71          }
72      }
73  }
