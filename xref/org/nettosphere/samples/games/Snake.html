32  package org.nettosphere.samples.games;
33  
34  import org.atmosphere.cpr.AtmosphereResource;
35  
36  import java.util.ArrayDeque;
37  import java.util.Collection;
38  import java.util.Deque;
39  
40  public class Snake {
41  
42      private static final int DEFAULT_LENGTH = 5;
43  
44      private final int id;
45      private final AtmosphereResource resource;
46  
47      private Direction direction;
48      private int length = DEFAULT_LENGTH;
49      private Location head;
50      private final Deque<Location> tail = new ArrayDeque<Location>();
51      private final String hexColor;
52  
53      public Snake(int id, AtmosphereResource resource) {
54          this.id = id;
55          this.hexColor = SnakeWebSocket.getRandomHexColor();
56          this.resource = resource;
57          resetState();
58      }
59  
60      private void resetState() {
61          this.direction = Direction.NONE;
62          this.head = SnakeWebSocket.getRandomLocation();
63          this.tail.clear();
64          this.length = DEFAULT_LENGTH;
65      }
66  
67      private synchronized void kill() {
68          resetState();
69          sendMessage("{'type': 'dead'}");
70      }
71  
72      private synchronized void reward() {
73          length++;
74          sendMessage("{'type': 'kill'}");
75      }
76  
77  
78      protected void sendMessage(String msg) {
79          resource.getResponse().write(msg);
80      }
81  
82      public synchronized void update(Collection<Snake> snakes) {
83          Location nextLocation = head.getAdjacentLocation(direction);
84          if (nextLocation.x >= SnakeWebSocket.PLAYFIELD_WIDTH) {
85              nextLocation.x = 0;
86          }
87          if (nextLocation.y >= SnakeWebSocket.PLAYFIELD_HEIGHT) {
88              nextLocation.y = 0;
89          }
90          if (nextLocation.x < 0) {
91              nextLocation.x = SnakeWebSocket.PLAYFIELD_WIDTH;
92          }
93          if (nextLocation.y < 0) {
94              nextLocation.y = SnakeWebSocket.PLAYFIELD_HEIGHT;
95          }
96          if (direction != Direction.NONE) {
97              tail.addFirst(head);
98              if (tail.size() > length) {
99                  tail.removeLast();
100             }
101             head = nextLocation;
102         }
103 
104         handleCollisions(snakes);
105     }
106 
107     private void handleCollisions(Collection<Snake> snakes) {
108         for (Snake snake : snakes) {
109             boolean headCollision = id != snake.id && snake.getHead().equals(head);
110             boolean tailCollision = snake.getTail().contains(head);
111             if (headCollision || tailCollision) {
112                 kill();
113                 if (id != snake.id) {
114                     snake.reward();
115                 }
116             }
117         }
118     }
119 
120     public synchronized Location getHead() {
121         return head;
122     }
123 
124     public synchronized Collection<Location> getTail() {
125         return tail;
126     }
127 
128     public synchronized void setDirection(Direction direction) {
129         this.direction = direction;
130     }
131 
132     public synchronized String getLocationsJson() {
133         StringBuilder sb = new StringBuilder();
134         sb.append(String.format("{x: %d, y: %d}",
135                 Integer.valueOf(head.x), Integer.valueOf(head.y)));
136         for (Location location : tail) {
137             sb.append(',');
138             sb.append(String.format("{x: %d, y: %d}",
139                     Integer.valueOf(location.x), Integer.valueOf(location.y)));
140         }
141         return String.format("{'id':%d,'body':[%s]}",
142                 Integer.valueOf(id), sb.toString());
143     }
144 
145     public int getId() {
146         return id;
147     }
148 
149     public String getHexColor() {
150         return hexColor;
151     }
152 }
