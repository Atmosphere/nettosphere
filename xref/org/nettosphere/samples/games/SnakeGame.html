31  package org.nettosphere.samples.games;
32  
33  import org.atmosphere.cpr.AtmosphereResource;
34  import org.atmosphere.cpr.BroadcasterFactory;
35  
36  import java.awt.*;
37  import java.io.IOException;
38  import java.util.Iterator;
39  import java.util.Random;
40  import java.util.concurrent.atomic.AtomicInteger;
41  
42  public class SnakeGame {
43      public static final int PLAYFIELD_WIDTH = 640;
44      public static final int PLAYFIELD_HEIGHT = 480;
45      public static final int GRID_SIZE = 10;
46  
47      protected static final AtomicInteger snakeIds = new AtomicInteger(0);
48      protected static final Random random = new Random();
49      protected final SnakeBroadcaster snakeBroadcaster;
50  
51      public SnakeGame() {
52          snakeBroadcaster = new SnakeBroadcaster(BroadcasterFactory.getDefault().lookup("/snake", true));
53      }
54  
55      public static String getRandomHexColor() {
56          float hue = random.nextFloat();
57          // sat between 0.1 and 0.3
58          float saturation = (random.nextInt(2000) + 1000) / 10000f;
59          float luminance = 0.9f;
60          Color color = Color.getHSBColor(hue, saturation, luminance);
61          return '#' + Integer.toHexString(
62                  (color.getRGB() & 0xffffff) | 0x1000000).substring(1);
63      }
64  
65      public static Location getRandomLocation() {
66          int x = roundByGridSize(random.nextInt(PLAYFIELD_WIDTH));
67          int y = roundByGridSize(random.nextInt(PLAYFIELD_HEIGHT));
68          return new Location(x, y);
69      }
70  
71      private static int roundByGridSize(int value) {
72          value = value + (GRID_SIZE / 2);
73          value = value / GRID_SIZE;
74          value = value * GRID_SIZE;
75          return value;
76      }
77  
78      public void onOpen(AtmosphereResource resource) throws IOException {
79          int id = snakeIds.getAndIncrement();
80          resource.session().setAttribute("id", id);
81          Snake snake = new Snake(id, resource);
82  
83          resource.session().setAttribute("snake", snake);
84          snakeBroadcaster.addSnake(snake);
85          StringBuilder sb = new StringBuilder();
86          for (Iterator<Snake> iterator = snakeBroadcaster.getSnakes().iterator();
87               iterator.hasNext(); ) {
88              snake = iterator.next();
89              sb.append(String.format("{id: %d, color: '%s'}",
90                      Integer.valueOf(snake.getId()), snake.getHexColor()));
91              if (iterator.hasNext()) {
92                  sb.append(',');
93              }
94          }
95          snakeBroadcaster.broadcast(String.format("{'type': 'join','data':[%s]}",
96                  sb.toString()));
97      }
98  
99      public void onClose(AtmosphereResource resource) {
100         snakeBroadcaster.removeSnake(snake(resource));
101         snakeBroadcaster.broadcast(String.format("{'type': 'leave', 'id': %d}",
102                 ((Integer) resource.session().getAttribute("id"))));
103     }
104 
105     protected Snake snake(AtmosphereResource resource) {
106         return (Snake) resource.session().getAttribute("snake");
107     }
108 
109     protected void onMessage(AtmosphereResource resource, String message) {
110         Snake snake = snake(resource);
111         if ("west".equals(message)) {
112             snake.setDirection(Direction.WEST);
113         } else if ("north".equals(message)) {
114             snake.setDirection(Direction.NORTH);
115         } else if ("east".equals(message)) {
116             snake.setDirection(Direction.EAST);
117         } else if ("south".equals(message)) {
118             snake.setDirection(Direction.SOUTH);
119         }
120     }
121 }
