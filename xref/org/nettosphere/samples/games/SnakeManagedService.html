16  package org.nettosphere.samples.games;
17  
18  import org.atmosphere.config.service.Disconnect;
19  import org.atmosphere.config.service.ManagedService;
20  import org.atmosphere.config.service.Post;
21  import org.atmosphere.config.service.Ready;
22  import org.atmosphere.cpr.AtmosphereRequest;
23  import org.atmosphere.cpr.AtmosphereResource;
24  import org.atmosphere.cpr.AtmosphereResourceEvent;
25  import org.atmosphere.cpr.AtmosphereResourceFactory;
26  import org.atmosphere.cpr.HeaderConfig;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.io.IOException;
31  import java.util.concurrent.ConcurrentLinkedQueue;
32  
33  @ManagedService(path = "/snake")
34  public class SnakeManagedService extends SnakeGame {
35  
36      private Logger logger = LoggerFactory.getLogger(SnakeManagedService.class);
37      private final ConcurrentLinkedQueue<String> uuids = new ConcurrentLinkedQueue<String>();
38  
39      @Ready
40      public void onReady(final AtmosphereResource r) {
41          if (!uuids.contains(r.uuid())) {
42              try {
43                  super.onOpen(r);
44              } catch (IOException e) {
45                  logger.error("", e);
46              }
47              uuids.add(r.uuid());
48          }
49      }
50  
51      @Disconnect
52      public void onDisconnect(AtmosphereResourceEvent event) {
53          AtmosphereRequest request = event.getResource().getRequest();
54          String s = request.getHeader(HeaderConfig.X_ATMOSPHERE_TRANSPORT);
55          if (s != null && s.equalsIgnoreCase(HeaderConfig.DISCONNECT)) {
56              SnakeManagedService.super.onClose(event.getResource());
57              uuids.remove(event.getResource().uuid());
58          }
59      }
60  
61  
62      @Post
63      public void onMessage(AtmosphereResource resource) {
64          try {
65              // Here we need to find the suspended AtmosphereResource
66              super.onMessage(AtmosphereResourceFactory.getDefault().find(resource.uuid()), resource.getRequest().getReader().readLine());
67          } catch (IOException e) {
68              e.printStackTrace();
69          }
70      }
71  
72  }
