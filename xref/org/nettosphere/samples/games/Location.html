32  package org.nettosphere.samples.games;
33  
34  public class Location {
35  
36      public int x;
37      public int y;
38  
39      public Location(int x, int y) {
40          this.x = x;
41          this.y = y;
42      }
43  
44      public Location getAdjacentLocation(Direction direction) {
45          switch (direction) {
46              case NORTH:
47                  return new Location(x, y - SnakeWebSocket.GRID_SIZE);
48              case SOUTH:
49                  return new Location(x, y + SnakeWebSocket.GRID_SIZE);
50              case EAST:
51                  return new Location(x + SnakeWebSocket.GRID_SIZE, y);
52              case WEST:
53                  return new Location(x - SnakeWebSocket.GRID_SIZE, y);
54              case NONE:
55                  // fall through
56              default:
57                  return this;
58          }
59      }
60  
61      @Override
62      public boolean equals(Object o) {
63          if (this == o) return true;
64          if (o == null || getClass() != o.getClass()) return false;
65  
66          Location location = (Location) o;
67  
68          if (x != location.x) return false;
69          if (y != location.y) return false;
70  
71          return true;
72      }
73  
74      @Override
75      public int hashCode() {
76          int result = x;
77          result = 31 * result + y;
78          return result;
79      }
80  }
