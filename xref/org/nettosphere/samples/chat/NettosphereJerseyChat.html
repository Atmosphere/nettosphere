16  package org.nettosphere.samples.chat;
17  
18  import org.atmosphere.nettosphere.Config;
19  import org.atmosphere.nettosphere.Nettosphere;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import java.io.BufferedReader;
24  import java.io.File;
25  import java.io.IOException;
26  import java.io.InputStreamReader;
27  
28  /**
29   * A bootstrap class that start Nettosphere and the Atmosphere Chat samples.
30   *
31  */
32  public class NettosphereJerseyChat {
33  
34      private static final Logger logger = LoggerFactory.getLogger(Nettosphere.class);
35  
36      public static void main(String[] args) throws IOException {
37          Config.Builder b = new Config.Builder();
38          //b.resource("./samples/jersey-chat/src/main/resources")
39          b.resource(new File(".").getAbsolutePath() + "/webapps")
40                  .resource(ResourceChat.class)
41                  .configFile("../conf/atmosphere.xml")
42                  .port(8080)
43                  .host("127.0.0.1")
44                  .initParam("org.atmosphere.websocket.messageContentType","application/json")
45                  .initParam("org.atmosphere.websocket.messageMethod","POST")
46                  .initParam("com.sun.jersey.api.json.POJOMappingFeature","true")
47                  .build();
48          Nettosphere s = new Nettosphere.Builder().config(b.build()).build();
49          s.start();
50          String a = "";
51  
52          logger.info("NettoSphere Jersey Chat Server started on port {}", 8080);
53          logger.info("Type quit to stop the server");
54          BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
55          while (!(a.equals("quit"))) {
56              a = br.readLine();
57          }
58          System.exit(-1);
59      }

61  }
