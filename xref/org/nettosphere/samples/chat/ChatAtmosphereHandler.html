16  package org.nettosphere.samples.chat;
17  
18  import org.atmosphere.config.service.ManagedService;
19  import org.atmosphere.cpr.AtmosphereResponse;
20  import org.atmosphere.handler.OnMessage;
21  import org.atmosphere.websocket.WebSocketEventListenerAdapter;
22  import org.codehaus.jackson.map.ObjectMapper;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import java.io.IOException;
27  import java.util.Date;
28  
29  /**
30   * Simple AtmosphereHandler that implement the logic to build a Chat application.
31   *
32   * @author Jeanfrancois Arcand
33   */
34  @ManagedService(path = "/chat"
35     /* Uncomment to receive connect/disconnect events for WebSocket */
36     /*, listeners = {ChatAtmosphereHandler.WebSocketEventListener.class} */)
37  public class ChatAtmosphereHandler extends OnMessage<String> {
38  
39      private final static Logger logger = LoggerFactory.getLogger(ChatAtmosphereHandler.class);
40      private final ObjectMapper mapper = new ObjectMapper();
41  
42      /**
43       * Echo the JSON Message we receives.
44       * @param response an {@link AtmosphereResponse}
45       * @param message  a message of type T
46       * @throws IOException
47       */
48      @Override
49      public void onMessage(AtmosphereResponse response, String message) throws IOException {
50          response.getWriter().write(mapper.writeValueAsString(mapper.readValue(message, Data.class)));
51      }
52  
53      /**
54       * Simple listener for events.
55       */
56      public final static class WebSocketEventListener extends WebSocketEventListenerAdapter {
57          @Override
58          public void onConnect(WebSocketEvent event) {
59              logger.debug("{}", event);
60          }
61  
62          @Override
63          public void onDisconnect(WebSocketEvent event) {
64              logger.debug("{}", event);
65          }
66      }
67  
68      public final static class Data {
69  
70          private String message;
71          private String author;
72          private long time;
73  
74          public Data() {
75              this("", "");
76          }
77  
78          public Data(String author, String message) {
79              this.author = author;
80              this.message = message;
81              this.time = new Date().getTime();
82          }
83  
84          public String getMessage() {
85              return message;
86          }
87  
88          public String getAuthor() {
89              return author;
90          }
91  
92          public void setAuthor(String author) {
93              this.author = author;
94          }
95  
96          public void setMessage(String message) {
97              this.message = message;
98          }
99  
100         public long getTime() {
101             return time;
102         }
103 
104         public void setTime(long time) {
105             this.time = time;
106         }
107 
108     }
109 }
