16  package org.nettosphere.samples.chat;
17  
18  import org.atmosphere.cpr.AtmosphereHandler;
19  import org.atmosphere.cpr.AtmosphereRequest;
20  import org.atmosphere.cpr.AtmosphereResource;
21  import org.atmosphere.cpr.AtmosphereResourceEvent;
22  import org.atmosphere.cpr.AtmosphereResponse;
23  
24  import java.io.IOException;
25  import java.util.Date;
26  
27  /**
28   * Simple AtmosphereHandler that implement the logic to build a Chat application.
29   *
30   * @author Jeanfrancois Arcand
31   */
32  public class ChatAtmosphereHandler implements AtmosphereHandler {
33  
34      @Override
35      public void onRequest(AtmosphereResource r) throws IOException {
36  
37          AtmosphereRequest req = r.getRequest();
38  
39          // First, tell Atmosphere to allow bi-directional communication by suspending.
40          if (req.getMethod().equalsIgnoreCase("GET")) {
41              // We are using HTTP long-polling with an invite timeout
42              r.suspend();
43          // Second, broadcast message to all connected users.
44          } else if (req.getMethod().equalsIgnoreCase("POST")) {
45              r.getBroadcaster().broadcast(req.getReader().readLine().trim());
46          }
47      }
48  
49      @Override
50      public void onStateChange(AtmosphereResourceEvent event) throws IOException {
51          AtmosphereResource r = event.getResource();
52          AtmosphereResponse res = r.getResponse();
53  
54          if (event.isSuspended()) {
55              String body = event.getMessage().toString();
56  
57              // Simple JSON -- Use Jackson for more complex structure
58              // Message looks like { "author" : "foo", "message" : "bar" }
59              String author = body.substring(body.indexOf(":") + 2, body.indexOf(",") - 1);
60              String message = body.substring(body.lastIndexOf(":") + 2, body.length() - 2);
61  
62              res.getWriter().write(new Data(author, message).toString());
63              switch (r.transport()) {
64                  case JSONP:
65                  case LONG_POLLING:
66                      event.getResource().resume();
67                      break;
68                  case WEBSOCKET :
69                  case STREAMING:
70                      res.getWriter().flush();
71                      break;
72              }
73          } else if (!event.isResuming()){
74              event.broadcaster().broadcast(new Data("Someone", "say bye bye!").toString());
75          }
76      }
77  
78      @Override
79      public void destroy() {
80      }
81  
82      private final static class Data {
83  
84          private final String text;
85          private final String author;
86  
87          public Data(String author, String text) {
88              this.author = author;
89              this.text = text;
90          }
91  
92          public String toString() {
93              return "{ \"text\" : \"" + text + "\", \"author\" : \"" + author + "\" , \"time\" : " + new Date().getTime() + "}";
94          }
95      }
96  }
