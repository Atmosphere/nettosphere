16  package org.nettosphere.samples.chat;
17  
18  import org.atmosphere.config.service.AtmosphereHandlerService;
19  import org.atmosphere.cpr.AtmosphereHandler;
20  import org.atmosphere.cpr.AtmosphereResource;
21  import org.atmosphere.cpr.AtmosphereResourceEvent;
22  import org.atmosphere.cpr.AtmosphereResponse;
23  import org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor;
24  import org.atmosphere.interceptor.BroadcastOnPostAtmosphereInterceptor;
25  
26  import java.io.IOException;
27  import java.util.Date;
28  
29  /**
30   * Simple SocketIO Implementation of the Atmosphere Chat
31   *
32   * @author Sebastien Dionne : sebastien.dionne@gmail.com
33   */
34  @AtmosphereHandlerService(path = "/chat", interceptors= {AtmosphereResourceLifecycleInterceptor.class
35  /*, BroadcastOnPostAtmosphereInterceptor.class*/})
36  public class SocketIOChatAtmosphereHandler implements AtmosphereHandler {
37  
38      @Override
39      public void onRequest(AtmosphereResource r) throws IOException {
40          r.getBroadcaster().broadcast(r.getRequest().getReader().readLine());
41      }
42  
43      @Override
44      public void onStateChange(AtmosphereResourceEvent event) throws IOException {
45          AtmosphereResource r = event.getResource();
46          AtmosphereResponse res = r.getResponse();
47  
48          if (event.isSuspended() && event.getMessage() != null) {
49              String body = event.getMessage().toString();
50  
51              // Simple JSON -- Use Jackson for more complex structure
52              // Message looks like { "author" : "foo", "message" : "bar" }
53              String author = body.substring(body.indexOf(":") + 2, body.indexOf(",") - 1);
54              String message = body.substring(body.lastIndexOf(":") + 2, body.length() - 2);
55  
56              res.getWriter().write(new Data(author, message).toString());
57          }
58      }
59  
60      @Override
61      public void destroy() {
62      }
63  
64      private final static class Data {
65  
66          private final String text;
67          private final String author;
68  
69          public Data(String author, String text) {
70              this.author = author;
71              this.text = text;
72          }
73  
74          public String toString() {
75              return "{ \"text\" : \"" + text + "\", \"author\" : \"" + author + "\" , \"time\" : " + new Date().getTime() + "}";
76          }
77      }
78  }
