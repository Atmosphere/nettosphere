16  package org.nettosphere.samples.chat;
17  
18  import org.atmosphere.config.service.ManagedService;
19  import org.atmosphere.config.service.Message;
20  import org.codehaus.jackson.map.ObjectMapper;
21  
22  import java.io.IOException;
23  import java.util.Date;
24  
25  @ManagedService(path = "/chat")
26  public class Chat {
27      private final ObjectMapper mapper = new ObjectMapper();
28  
29      @Message
30      public String onMessage(String message) throws IOException {
31          return mapper.writeValueAsString(mapper.readValue(message, Data.class));
32      }
33  
34      public final static class Data {
35  
36          private String message;
37          private String author;
38          private long time;
39  
40          public Data() {
41              this("", "");
42          }
43  
44          public Data(String author, String message) {
45              this.author = author;
46              this.message = message;
47              this.time = new Date().getTime();
48          }
49  
50          public String getMessage() {
51              return message;
52          }
53  
54          public String getAuthor() {
55              return author;
56          }
57  
58          public void setAuthor(String author) {
59              this.author = author;
60          }
61  
62          public void setMessage(String message) {
63              this.message = message;
64          }
65  
66          public long getTime() {
67              return time;
68          }
69  
70          public void setTime(long time) {
71              this.time = time;
72          }
73  
74      }
75  }
